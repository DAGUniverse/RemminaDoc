Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

79 static gboolean onMainThread_cb(struct onMainThread_cb_data *d)
80 {
81  TRACE_CALL(__func__);
82  if ( !d->cancelled ) {
83  switch ( d->func ) {
84  case FUNC_GTK_SOCKET_ADD_ID:
85  gtk_socket_add_id( d->sk, d->w );
86  break;
87  }
88  pthread_mutex_unlock( &d->mu );
89  } else {
90  /* thread has been cancelled, so we must free d memory here */
91  g_free( d );
92  }
93  return G_SOURCE_REMOVE;
94 }

105 static void onMainThread_schedule_callback_and_wait( struct onMainThread_cb_data *d )
106 {
107  TRACE_CALL(__func__);
108  d->cancelled = FALSE;
109  pthread_cleanup_push( onMainThread_cleanup_handler, d );
110  pthread_mutex_init( &d->mu, NULL );
111  pthread_mutex_lock( &d->mu );
112  gdk_threads_add_idle( (GSourceFunc)onMainThread_cb, (gpointer)d );
113 
114  pthread_mutex_lock( &d->mu );
115 
116  pthread_cleanup_pop(0);
117  pthread_mutex_unlock( &d->mu );
118  pthread_mutex_destroy( &d->mu );
119 }

141 static gboolean remmina_plugin_nx_try_window_id(Window window_id)
142 {
143  TRACE_CALL(__func__);
144  gint i;
145  gboolean found = FALSE;
146 
147  pthread_mutex_lock(&remmina_nx_init_mutex);
148  for (i = 0; i < remmina_nx_window_id_array->len; i++) {
149  if (g_array_index(remmina_nx_window_id_array, Window, i) == window_id) {
150  found = TRUE;
151  break;
152  }
153  }
154  if (!found) {
155  g_array_append_val(remmina_nx_window_id_array, window_id);
156  }
157  pthread_mutex_unlock(&remmina_nx_init_mutex);
158 
159  return (!found);
160 }

239 static gboolean remmina_plugin_nx_monitor_create_notify(RemminaProtocolWidget *gp, const gchar *cmd)
240 {
241  TRACE_CALL(__func__);
242  RemminaPluginNxData *gpdata;
243  Atom atom;
244  XEvent xev;
245  Window w;
246  Atom type;
247  int format;
248  unsigned long nitems, rest;
249  unsigned char *data = NULL;
250  struct timespec ts;
251 
252  CANCEL_DEFER
253 
254  gpdata = GET_PLUGIN_DATA(gp);
255  atom = XInternAtom(gpdata->display, "WM_COMMAND", True);
256  if (atom == None)
257  return FALSE;
258 
259  ts.tv_sec = 0;
260  ts.tv_nsec = 200000000;
261 
262  while (1) {
263  pthread_testcancel();
264  while (!XPending(gpdata->display)) {
265  nanosleep(&ts, NULL);
266  continue;
267  }
268  XNextEvent(gpdata->display, &xev);
269  if (xev.type != CreateNotify)
270  continue;
271  w = xev.xcreatewindow.window;
272  if (XGetWindowProperty(gpdata->display, w, atom, 0, 255, False, AnyPropertyType, &type, &format, &nitems, &rest,
273  &data) != Success)
274  continue;
275  if (data && strstr((char*)data, cmd) && remmina_plugin_nx_try_window_id(w)) {
276  gpdata->window_id = w;
277  XFree(data);
278  break;
279  }
280  if (data)
281  XFree(data);
282  }
283 
284  XSetErrorHandler(gpdata->orig_handler);
285  XCloseDisplay(gpdata->display);
286  gpdata->display = NULL;
287 
288  CANCEL_ASYNC
289  return TRUE;
290 }

306 static gboolean remmina_plugin_nx_start_session(RemminaProtocolWidget *gp)
307 {
308  TRACE_CALL(__func__);
309  RemminaPluginNxData *gpdata = GET_PLUGIN_DATA(gp);
310  RemminaFile *remminafile;
311  RemminaNXSession *nx;
312  const gchar *type, *app;
313  gchar *s1, *s2;
314  gint port;
315  gint ret;
316  gboolean is_empty_list;
317  gint event_type = 0;
318  const gchar *cs;
319  gint i;
320  gboolean disablepasswordstoring;
321 
322  remminafile = remmina_plugin_nx_service->protocol_plugin_get_file(gp);
323  nx = gpdata->nx;
324 
325  /* Connect */
326 
327  remmina_nx_session_set_encryption(nx,
328  remmina_plugin_nx_service->file_get_int(remminafile, "disableencryption", FALSE) ? 0 : 1);
329  remmina_nx_session_set_localport(nx, remmina_plugin_nx_service->pref_get_sshtunnel_port());
330  remmina_nx_session_set_log_callback(nx, remmina_plugin_nx_service->log_printf);
331 
332  s2 = remmina_plugin_nx_service->protocol_plugin_start_direct_tunnel(gp, 22, FALSE);
333  if (s2 == NULL) {
334  return FALSE;
335  }
336  remmina_plugin_nx_service->get_server_port(s2, 22, &s1, &port);
337  g_free(s2);
338 
339  if (!remmina_nx_session_open(nx, s1, port, remmina_plugin_nx_service->file_get_string(remminafile, "nx_privatekey"),
340  remmina_plugin_nx_ssh_auth_callback, gp)) {
341  g_free(s1);
342  return FALSE;
343  }
344  g_free(s1);
345 
346  /* Login */
347 
348  s1 = g_strdup(remmina_plugin_nx_service->file_get_string(remminafile, "username"));
349  s2 = g_strdup(remmina_plugin_nx_service->file_get_string(remminafile, "password"));
350 
351  if (s1 && s2) {
352  ret = remmina_nx_session_login(nx, s1, s2);
353  }else {
354  g_free(s1);
355  g_free(s2);
356 
357  disablepasswordstoring = remmina_plugin_nx_service->file_get_int(remminafile, "disablepasswordstoring", FALSE);
358  ret = remmina_plugin_nx_service->protocol_plugin_init_authuserpwd(gp, FALSE, !disablepasswordstoring);
359 
360  if (ret != GTK_RESPONSE_OK)
361  return FALSE;
362 
363  s1 = remmina_plugin_nx_service->protocol_plugin_init_get_username(gp);
364  s2 = remmina_plugin_nx_service->protocol_plugin_init_get_password(gp);
365  ret = remmina_nx_session_login(nx, s1, s2);
366  }
367  g_free(s1);
368  g_free(s2);
369 
370  if (!ret)
371  return FALSE;
372 
373  remmina_plugin_nx_service->protocol_plugin_init_save_cred(gp);
374 
375  /* Prepare the session type and application */
376  cs = remmina_plugin_nx_service->file_get_string(remminafile, "exec");
377  if (!cs || g_strcmp0(cs, "GNOME") == 0) {
378  type = "unix-gnome";
379  app = NULL;
380  }else
381  if (g_strcmp0(cs, "KDE") == 0) {
382  type = "unix-kde";
383  app = NULL;
384  }else
385  if (g_strcmp0(cs, "Xfce") == 0) {
386  /* NX does not know Xfce. So we simply launch the Xfce startup program. */
387  type = "unix-application";
388  app = "startxfce4";
389  }else
390  if (g_strcmp0(cs, "Shadow") == 0) {
391  type = "shadow";
392  app = NULL;
393  }else {
394  type = "unix-application";
395  app = cs;
396  }
397 
398  /* List sessions */
399 
400  gpdata->attach_session = (g_strcmp0(type, "shadow") == 0);
401  while (1) {
402  remmina_nx_session_add_parameter(nx, "type", type);
403  if (!gpdata->attach_session) {
404  remmina_nx_session_add_parameter(nx, "user",
405  remmina_plugin_nx_service->file_get_string(remminafile, "username"));
406  remmina_nx_session_add_parameter(nx, "status", "suspended,running");
407  }
408 
409  if (!remmina_nx_session_list(nx)) {
410  return FALSE;
411  }
412 
413  is_empty_list = !remmina_nx_session_iter_first(nx, &gpdata->iter);
414  if (is_empty_list && !gpdata->manager_started && !gpdata->attach_session) {
415  event_type = REMMINA_NX_EVENT_START;
416  }else {
417  remmina_nx_session_manager_start(gp);
418  event_type = remmina_plugin_nx_wait_signal(gpdata);
419  if (event_type == REMMINA_NX_EVENT_CANCEL) {
420  return FALSE;
421  }
422  if (event_type == REMMINA_NX_EVENT_TERMINATE) {
423  if (!is_empty_list) {
424  s1 = remmina_nx_session_iter_get(nx, &gpdata->iter, REMMINA_NX_SESSION_COLUMN_ID);
425  remmina_nx_session_add_parameter(nx, "sessionid", s1);
426  g_free(s1);
427  if (!remmina_nx_session_terminate(nx)) {
428  remmina_nx_session_manager_start(gp);
429  remmina_plugin_nx_wait_signal(gpdata);
430  }
431  }
432  continue;
433  }
434  }
435 
436  break;
437  }
438 
439  /* Start, Restore or Attach, based on the setting and existing session */
440  remmina_nx_session_add_parameter(nx, "type", type);
441  i = remmina_plugin_nx_service->file_get_int(remminafile, "quality", 0);
442  remmina_nx_session_add_parameter(nx, "link", i > 2 ? "lan" : i == 2 ? "adsl" : i == 1 ? "isdn" : "modem");
443  remmina_nx_session_add_parameter(nx, "geometry", "%ix%i",
444  remmina_plugin_nx_service->get_profile_remote_width(gp),
445  remmina_plugin_nx_service->get_profile_remote_height(gp));
446  remmina_nx_session_add_parameter(nx, "keyboard", remmina_kbtype);
447  remmina_nx_session_add_parameter(nx, "client", "linux");
448  remmina_nx_session_add_parameter(nx, "media", "0");
449  remmina_nx_session_add_parameter(nx, "clipboard",
450  remmina_plugin_nx_service->file_get_int(remminafile, "disableclipboard", FALSE) ? "none" : "both");
451 
452  switch (event_type) {
453 
454  case REMMINA_NX_EVENT_START:
455  if (app)
456  remmina_nx_session_add_parameter(nx, "application", app);
457 
458  remmina_nx_session_add_parameter(nx, "session",
459  remmina_plugin_nx_service->file_get_string(remminafile, "name"));
460  remmina_nx_session_add_parameter(nx, "screeninfo", "%ix%ix24+render",
461  remmina_plugin_nx_service->file_get_int(remminafile, "resolution_width", 0),
462  remmina_plugin_nx_service->file_get_int(remminafile, "resolution_height", 0));
463 
464  if (!remmina_nx_session_start(nx))
465  return FALSE;
466  break;
467 
468  case REMMINA_NX_EVENT_ATTACH:
469  s1 = remmina_nx_session_iter_get(nx, &gpdata->iter, REMMINA_NX_SESSION_COLUMN_ID);
470  remmina_nx_session_add_parameter(nx, "id", s1);
471  g_free(s1);
472 
473  s1 = remmina_nx_session_iter_get(nx, &gpdata->iter, REMMINA_NX_SESSION_COLUMN_DISPLAY);
474  remmina_nx_session_add_parameter(nx, "display", s1);
475  g_free(s1);
476 
477  if (!remmina_nx_session_attach(nx))
478  return FALSE;
479  break;
480 
481  case REMMINA_NX_EVENT_RESTORE:
482  s1 = remmina_nx_session_iter_get(nx, &gpdata->iter, REMMINA_NX_SESSION_COLUMN_ID);
483  remmina_nx_session_add_parameter(nx, "id", s1);
484  g_free(s1);
485 
486  remmina_nx_session_add_parameter(nx, "session",
487  remmina_plugin_nx_service->file_get_string(remminafile, "name"));
488 
489  if (!remmina_nx_session_restore(nx))
490  return FALSE;
491  break;
492 
493  default:
494  return FALSE;
495  }
496 
497  if (!remmina_nx_session_tunnel_open(nx))
498  return FALSE;
499 
500  if (!remmina_plugin_nx_start_create_notify(gp))
501  return FALSE;
502 
503  /* nxproxy */
504  if (!remmina_nx_session_invoke_proxy(nx, -1, remmina_plugin_nx_on_proxy_exit, gp))
505  return FALSE;
506 
507  /* get the window id of the remote nxagent */
508  if (!remmina_plugin_nx_monitor_create_notify(gp, "nxagent"))
509  return FALSE;
510 
511  /* embed it */
512  onMainThread_gtk_socket_add_id(GTK_SOCKET(gpdata->socket), gpdata->window_id);
513 
514 
515  return TRUE;
516 }
