Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

66 static gpointer
67 remmina_plugin_sftp_main_thread(gpointer data)
68 {
69  TRACE_CALL(__func__);
70  RemminaProtocolWidget *gp = (RemminaProtocolWidget*)data;
71  RemminaPluginSftpData *gpdata;
72  RemminaFile *remminafile;
73  RemminaSSH *ssh;
74  RemminaSFTP *sftp = NULL;
75  gboolean cont = FALSE;
76  gint ret;
77  const gchar *cs;
78  const gchar *saveserver;
79  const gchar *saveusername;
80  gchar *hostport;
81  gchar tunneluser[33]; /* On linux a username can have a 32 char lenght */
82  gchar tunnelserver[256]; /* On linux a servername can have a 255 char lenght */
83  gchar tunnelport[6]; /* A TCP port can have a maximum value of 65535 */
84  gchar *host;
85  gint port;
86 
87  pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, NULL);
88  CANCEL_ASYNC
89 
90  gpdata = GET_PLUGIN_DATA(gp);
91 
92  /* remmina_plugin_service->protocol_plugin_start_direct_tunnel start the
93  * SSH Tunnel and return the server + port string
94  * Therefore we set the SSH Tunnel username here, before the tunnel
95  * is established and than set it back to the destination SSH user.
96  *
97  * */
98  remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
99  /* We save the ssh server name, so that we can restore it at the end of the connection */
100  saveserver = remmina_plugin_service->file_get_string(remminafile, "ssh_server");
101  remmina_plugin_service->file_set_string(remminafile, "save_ssh_server", g_strdup(saveserver));
102  /* We save the ssh username, so that we can restore it at the end of the connection */
103  saveusername = remmina_plugin_service->file_get_string(remminafile, "ssh_username");
104  remmina_plugin_service->file_set_string(remminafile, "save_ssh_username", g_strdup(saveusername));
105 
106  if (saveserver) {
107  /* if the server string contains the character @ we extract the user
108  * and the server string (host:port)
109  * */
110  if (strchr(saveserver, '@')) {
111  sscanf(saveserver, "%[_a-zA-Z0-9.]@%[_a-zA-Z0-9.]:%[0-9]",
112  tunneluser, tunnelserver, tunnelport);
113  g_print ("Username: %s, tunneluser: %s\n",
114  remmina_plugin_service->file_get_string(remminafile, "ssh_username"), tunneluser);
115  if (saveusername != NULL && tunneluser[0] != '\0') {
116  remmina_plugin_service->file_set_string(remminafile, "ssh_username", NULL);
117  }
118  remmina_plugin_service->file_set_string(remminafile, "ssh_username",
119  g_strdup(tunneluser));
120  remmina_plugin_service->file_set_string(remminafile, "ssh_server",
121  g_strconcat (tunnelserver, ":", tunnelport, NULL));
122  }
123  }
124 
125  hostport = remmina_plugin_service->protocol_plugin_start_direct_tunnel(gp, 22, FALSE);
126  /* We restore the ssh username as the tunnel is set */
127  remmina_plugin_service->file_set_string(remminafile, "ssh_username", g_strdup(saveusername));
128  if (hostport == NULL) {
129  return FALSE;
130  }
131  remmina_plugin_service->get_server_port(hostport, 22, &host, &port);
132 
133  ssh = g_object_get_data(G_OBJECT(gp), "user-data");
134  if (ssh) {
135  /* Create SFTP connection based on existing SSH session */
136  sftp = remmina_sftp_new_from_ssh(ssh);
137  if (remmina_ssh_init_session(REMMINA_SSH(sftp)) &&
138  remmina_ssh_auth(REMMINA_SSH(sftp), NULL) > 0 &&
139  remmina_sftp_open(sftp)) {
140  cont = TRUE;
141  }
142  }else {
143  /* New SFTP connection */
144  if (remmina_plugin_service->file_get_int(remminafile, "ssh_enabled", FALSE)) {
145  remmina_plugin_service->file_set_string(remminafile, "ssh_server", g_strdup(hostport));
146  }else {
147  remmina_plugin_service->file_set_string(remminafile, "ssh_server",
148  remmina_plugin_service->file_get_string(remminafile, "server"));
149  }
150  g_free(hostport);
151  g_free(host);
152 
153  sftp = remmina_sftp_new_from_file(remminafile);
154  while (1) {
155  if (!remmina_ssh_init_session(REMMINA_SSH(sftp))) {
156  remmina_plugin_service->protocol_plugin_set_error(gp, "%s", REMMINA_SSH(sftp)->error);
157  break;
158  }
159 
160  ret = remmina_ssh_auth_gui(REMMINA_SSH(sftp),
161  REMMINA_INIT_DIALOG(remmina_protocol_widget_get_init_dialog(gp)),
162  remminafile);
163  if (ret == 0) {
164  remmina_plugin_service->protocol_plugin_set_error(gp, "%s", REMMINA_SSH(sftp)->error);
165  }
166  if (ret <= 0) break;
167 
168  if (!remmina_sftp_open(sftp)) {
169  remmina_plugin_service->protocol_plugin_set_error(gp, "%s", REMMINA_SSH(sftp)->error);
170  break;
171  }
172 
173  cs = remmina_plugin_service->file_get_string(remminafile, "execpath");
174  if (cs && cs[0]) {
175  remmina_ftp_client_set_dir(REMMINA_FTP_CLIENT(gpdata->client), cs);
176  }
177 
178  cont = TRUE;
179  break;
180  }
181 
182  /* We restore the ssh_server name */
183  remmina_plugin_service->file_set_string(remminafile, "ssh_server",
184  remmina_plugin_service->file_get_string(remminafile, "save_ssh_server"));
185  }
186  if (!cont) {
187  if (sftp) remmina_sftp_free(sftp);
188  IDLE_ADD((GSourceFunc)remmina_plugin_service->protocol_plugin_close_connection, gp);
189  return NULL;
190  }
191 
192  remmina_sftp_client_open(REMMINA_SFTP_CLIENT(gpdata->client), sftp);
193  /* RemminaSFTPClient owns the object, we just take the reference */
194  gpdata->sftp = sftp;
195 
196  remmina_plugin_service->protocol_plugin_emit_signal(gp, "connect");
197 
198  gpdata->thread = 0;
199  return NULL;
200 }

292 static void
293 remmina_plugin_sftp_call_feature(RemminaProtocolWidget *gp, const RemminaProtocolFeature *feature)
294 {
295  TRACE_CALL(__func__);
296  RemminaPluginSftpData *gpdata = GET_PLUGIN_DATA(gp);
297  RemminaFile *remminafile;
298 
299  remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
300  switch (feature->id) {
301  case REMMINA_PROTOCOL_FEATURE_TOOL_SSH:
302  remmina_plugin_service->open_connection(
303  remmina_file_dup_temp_protocol(remmina_plugin_service->protocol_plugin_get_file(gp), "SSH"),
304  NULL, gpdata->sftp, NULL);
305  return;
306  case REMMINA_PROTOCOL_FEATURE_TOOL_SFTP:
307  remmina_plugin_service->open_connection(
308  remmina_file_dup_temp_protocol(remmina_plugin_service->protocol_plugin_get_file(gp), "SFTP"),
309  NULL, gpdata->sftp, NULL);
310  return;
311  case REMMINA_PLUGIN_SFTP_FEATURE_PREF_SHOW_HIDDEN:
312  remmina_ftp_client_set_show_hidden(REMMINA_FTP_CLIENT(gpdata->client),
313  remmina_plugin_service->file_get_int(remminafile, "showhidden", FALSE));
314  return;
315  case REMMINA_PLUGIN_SFTP_FEATURE_PREF_OVERWRITE_ALL:
316  remmina_ftp_client_set_overwrite_status(REMMINA_FTP_CLIENT(gpdata->client),
317  remmina_plugin_service->file_get_int(remminafile,
318  REMMINA_PLUGIN_SFTP_FEATURE_PREF_OVERWRITE_ALL_KEY, FALSE));
319  return;
320  }
