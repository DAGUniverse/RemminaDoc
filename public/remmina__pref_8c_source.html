Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

69 static void remmina_pref_gen_secret(void)
70 {
71  TRACE_CALL(__func__);
72  guchar s[32];
73  gint i;
74  GTimeVal gtime;
75  GKeyFile *gkeyfile;
76  gchar *content;
77  gsize length;
78 
79  g_get_current_time(&gtime);
80  srand(gtime.tv_sec);
81 
82  for (i = 0; i < 32; i++) {
83  s[i] = (guchar)(rand() % 256);
84  }
85  remmina_pref.secret = g_base64_encode(s, 32);
86 
87  gkeyfile = g_key_file_new();
88  g_key_file_load_from_file(gkeyfile, remmina_pref_file, G_KEY_FILE_NONE, NULL);
89  g_key_file_set_string(gkeyfile, "remmina_pref", "secret", remmina_pref.secret);
90  content = g_key_file_to_data(gkeyfile, &length, NULL);
91  g_file_set_contents(remmina_pref_file, content, length, NULL);
92 
93  g_key_file_free(gkeyfile);
94  g_free(content);
95 }

113 static void remmina_pref_init_keymap(void)
114 {
115  TRACE_CALL(__func__);
116  GKeyFile *gkeyfile;
117  gchar **groups;
118  gchar **gptr;
119  gchar **keys;
120  gchar **kptr;
121  gsize nkeys;
122  gchar *value;
123  guint *table;
124  guint *tableptr;
125  guint k1, k2;
126 
127  if (remmina_keymap_table)
128  g_hash_table_destroy(remmina_keymap_table);
129  remmina_keymap_table = g_hash_table_new_full(g_str_hash, g_str_equal, g_free, g_free);
130 
131  gkeyfile = g_key_file_new();
132  if (!g_key_file_load_from_file(gkeyfile, remmina_keymap_file, G_KEY_FILE_NONE, NULL)) {
133  if (!g_key_file_load_from_data(gkeyfile, default_keymap_data, strlen(default_keymap_data), G_KEY_FILE_NONE,
134  NULL)) {
135  g_print("Failed to initialize keymap table\n");
136  g_key_file_free(gkeyfile);
137  return;
138  }
139  }
140 
141  groups = g_key_file_get_groups(gkeyfile, NULL);
142  gptr = groups;
143  while (*gptr) {
144  keys = g_key_file_get_keys(gkeyfile, *gptr, &nkeys, NULL);
145  table = g_new0(guint, nkeys * 2 + 1);
146  g_hash_table_insert(remmina_keymap_table, g_strdup(*gptr), table);
147 
148  kptr = keys;
149  tableptr = table;
150  while (*kptr) {
151  k1 = remmina_pref_get_keyval_from_str(*kptr);
152  if (k1) {
153  value = g_key_file_get_string(gkeyfile, *gptr, *kptr, NULL);
154  k2 = remmina_pref_get_keyval_from_str(value);
155  g_free(value);
156  *tableptr++ = k1;
157  *tableptr++ = k2;
158  }
159  kptr++;
160  }
161  g_strfreev(keys);
162  gptr++;
163  }
164  g_strfreev(groups);
165  g_key_file_free(gkeyfile);
166 }

169 static gboolean remmina_pref_file_do_copy(const char *src_path, const char *dst_path)
170 {
171  GFile *src = g_file_new_for_path(src_path), *dst = g_file_new_for_path(dst_path);
172  /* We don't overwrite the target if it exists, because overwrite is not set */
173  const gboolean ok = g_file_copy(src, dst, G_FILE_COPY_NONE, NULL, NULL, NULL, NULL);
174 
175  g_object_unref(dst);
176  g_object_unref(src);
177 
178  return ok;
179 }
