Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

209 static void remmina_plugin_spice_file_transfer_finished_cb(SpiceFileTransferTask *task, GError *error, RemminaProtocolWidget *gp)
210 {
211  TRACE_CALL(__func__);
212 
213  gchar *filename, *notification_message;
214  GNotification *notification;
215  RemminaPluginSpiceData *gpdata = GET_PLUGIN_DATA(gp);
216 
217  /*
218  * Send a desktop notification to inform about the outcome of
219  * the file transfer.
220  */
221  filename = spice_file_transfer_task_get_filename(task);
222 
223  if (error)
224  {
225  notification = g_notification_new(_("Transfer error"));
226  notification_message = g_strdup_printf(_("%s: %s"),
227  filename, error->message);
228  }
229  else
230  {
231  notification = g_notification_new(_("Transfer completed"));
232  notification_message = g_strdup_printf(_("File %s transferred successfully"),
233  filename);
234  }
235 
236  g_notification_set_body(notification, notification_message);
237  g_application_send_notification(g_application_get_default(),
238  "remmina-plugin-spice-file-transfer-finished",
239  notification);
240 
241  g_hash_table_remove(gpdata->file_transfers, task);
242 
243  if (!g_hash_table_size(gpdata->file_transfers))
244  {
245  gtk_widget_hide(gpdata->file_transfer_dialog);
246  }
247 
248  g_free(filename);
249  g_free(notification_message);
250  g_object_unref(notification);
251 }
