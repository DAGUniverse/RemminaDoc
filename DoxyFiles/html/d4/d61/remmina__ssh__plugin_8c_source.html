Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

206 static gpointer
207 remmina_plugin_ssh_main_thread (gpointer data)
208 {
209  TRACE_CALL("remmina_plugin_ssh_main_thread");
210  RemminaProtocolWidget *gp = (RemminaProtocolWidget*) data;
211  RemminaPluginSshData *gpdata;
212  RemminaFile *remminafile;
213  RemminaSSH *ssh;
214  RemminaSSHShell *shell = NULL;
215  gboolean cont = FALSE;
216  gint ret;
217  gchar *charset;
218 
219  pthread_setcancelstate (PTHREAD_CANCEL_ENABLE, NULL);
220  CANCEL_ASYNC
221 
222  gpdata = GET_PLUGIN_DATA(gp);
223 
224  ssh = g_object_get_data (G_OBJECT(gp), "user-data");
225  if (ssh)
226  {
227  /* Create SSH Shell connection based on existing SSH session */
228  shell = remmina_ssh_shell_new_from_ssh (ssh);
229  if (remmina_ssh_init_session (REMMINA_SSH (shell)) &&
230  remmina_ssh_auth (REMMINA_SSH (shell), NULL) > 0 &&
231  remmina_ssh_shell_open (shell, (RemminaSSHExitFunc)
232  remmina_plugin_service->protocol_plugin_close_connection, gp))
233  {
234  cont = TRUE;
235  }
236  }
237  else
238  {
239  /* New SSH Shell connection */
240  remminafile = remmina_plugin_service->protocol_plugin_get_file (gp);
241  remmina_plugin_service->file_set_string (remminafile, "ssh_server",
242  remmina_plugin_service->file_get_string (remminafile, "server"));
243 
244  shell = remmina_ssh_shell_new_from_file (remminafile);
245  while (1)
246  {
247  if (!remmina_ssh_init_session (REMMINA_SSH (shell)))
248  {
249  remmina_plugin_service->protocol_plugin_set_error (gp, "%s", REMMINA_SSH (shell)->error);
250  break;
251  }
252 
253  ret = remmina_ssh_auth_gui (REMMINA_SSH (shell),
254  REMMINA_INIT_DIALOG (remmina_protocol_widget_get_init_dialog (gp)),
255  remminafile);
256  if (ret == 0)
257  {
258  remmina_plugin_service->protocol_plugin_set_error (gp, "%s", REMMINA_SSH (shell)->error);
259  }
260  if (ret <= 0) break;
261 
262  if (!remmina_ssh_shell_open (shell, (RemminaSSHExitFunc)
263  remmina_plugin_service->protocol_plugin_close_connection, gp))
264  {
265  remmina_plugin_service->protocol_plugin_set_error (gp, "%s", REMMINA_SSH (shell)->error);
266  break;
267  }
268 
269  cont = TRUE;
270  break;
271  }
272  }
273  if (!cont)
274  {
275  if (shell) remmina_ssh_shell_free (shell);
276  IDLE_ADD ((GSourceFunc) remmina_plugin_service->protocol_plugin_close_connection, gp);
277  return NULL;
278  }
279 
280  gpdata->shell = shell;
281 
282  charset = REMMINA_SSH (shell)->charset;
283  remmina_plugin_ssh_vte_terminal_set_encoding_and_pty(VTE_TERMINAL (gpdata->vte), charset, shell->master, shell->slave);
284  remmina_plugin_service->protocol_plugin_emit_signal (gp, "connect");
285 
286  gpdata->thread = 0;
287  return NULL;
288 }

290 void remmina_plugin_ssh_vte_terminal_set_encoding_and_pty(VteTerminal *terminal, const char *codeset, int master, int slave)
291 {
292  TRACE_CALL("remmina_plugin_ssh_vte_terminal_set_encoding_and_pty");
293  if ( !remmina_masterthread_exec_is_main_thread() )
294  {
295  /* Allow the execution of this function from a non main thread */
296  RemminaMTExecData *d;
297  d = (RemminaMTExecData*)g_malloc( sizeof(RemminaMTExecData) );
298  d->func = FUNC_VTE_TERMINAL_SET_ENCODING_AND_PTY;
299  d->p.vte_terminal_set_encoding_and_pty.terminal = terminal;
300  d->p.vte_terminal_set_encoding_and_pty.codeset = codeset;
301  d->p.vte_terminal_set_encoding_and_pty.master = master;
302  remmina_masterthread_exec_and_wait(d);
303  g_free(d);
304  return;
305  }
306 
307  setlocale(LC_ALL, "");
308  if (codeset && codeset[0] != '\0')
309  {
310 #if VTE_CHECK_VERSION(0,38,0)
311  vte_terminal_set_encoding (terminal, codeset, NULL);
312 #else
313  vte_terminal_set_emulation(terminal, "xterm");
314  vte_terminal_set_encoding (terminal, codeset);
315 #endif
316  }
317 
318  vte_terminal_set_backspace_binding(terminal, VTE_ERASE_ASCII_DELETE);
319  vte_terminal_set_delete_binding(terminal, VTE_ERASE_DELETE_SEQUENCE);
320 
321 #if VTE_CHECK_VERSION(0,38,0)
322  /* vte_pty_new_foreig expect master FD, see https://bugzilla.gnome.org/show_bug.cgi?id=765382 */
323  vte_terminal_set_pty (terminal, vte_pty_new_foreign_sync(master, NULL, NULL));
324 #else
325  vte_terminal_set_pty (terminal, master);
326 #endif
327 
328 }
