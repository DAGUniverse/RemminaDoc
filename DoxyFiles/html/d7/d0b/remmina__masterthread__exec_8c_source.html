Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

46 static gboolean remmina_masterthread_exec_callback(RemminaMTExecData *d)
47 {
48 
49  /* This function is called on main GTK Thread via gdk_threads_add_idlde()
50  * from remmina_masterthread_exec_and_wait() */
51 
52  if (!d->cancelled)
53  {
54  switch(d->func)
55  {
56  case FUNC_INIT_SAVE_CRED:
57  remmina_protocol_widget_init_save_cred(d->p.init_save_creds.gp);
58  break;
59  case FUNC_CHAT_RECEIVE:
60  remmina_protocol_widget_chat_receive(d->p.chat_receive.gp, d->p.chat_receive.text);
61  break;
62  case FUNC_FILE_GET_STRING:
63  d->p.file_get_string.retval = remmina_file_get_string( d->p.file_get_string.remminafile, d->p.file_get_string.setting );
64  break;
65  case FUNC_DIALOG_SERVERKEY_CONFIRM:
66  d->p.dialog_serverkey_confirm.retval = remmina_init_dialog_serverkey_confirm( d->p.dialog_serverkey_confirm.dialog,
67  d->p.dialog_serverkey_confirm.serverkey, d->p.dialog_serverkey_confirm.prompt );
68  break;
69  case FUNC_DIALOG_AUTHPWD:
70  d->p.dialog_authpwd.retval = remmina_init_dialog_authpwd(d->p.dialog_authpwd.dialog,
71  d->p.dialog_authpwd.label, d->p.dialog_authpwd.allow_save);
72  break;
73  case FUNC_GTK_LABEL_SET_TEXT:
74  gtk_label_set_text( d->p.gtk_label_set_text.label, d->p.gtk_label_set_text.str );
75  break;
76  case FUNC_DIALOG_AUTHUSERPWD:
77  d->p.dialog_authuserpwd.retval = remmina_init_dialog_authuserpwd( d->p.dialog_authuserpwd.dialog,
78  d->p.dialog_authuserpwd.want_domain, d->p.dialog_authuserpwd.default_username,
79  d->p.dialog_authuserpwd.default_domain, d->p.dialog_authuserpwd.allow_save );
80  break;
81  case FUNC_DIALOG_CERT:
82  d->p.dialog_certificate.retval = remmina_init_dialog_certificate( d->p.dialog_certificate.dialog,
83  d->p.dialog_certificate.subject, d->p.dialog_certificate.issuer, d->p.dialog_certificate.fingerprint );
84  break;
85  case FUNC_DIALOG_CERTCHANGED:
86  d->p.dialog_certchanged.retval = remmina_init_dialog_certificate_changed( d->p.dialog_certchanged.dialog,
87  d->p.dialog_certchanged.subject, d->p.dialog_certchanged.issuer, d->p.dialog_certchanged.new_fingerprint,
88  d->p.dialog_certchanged.old_fingerprint );
89  break;
90  case FUNC_DIALOG_AUTHX509:
91  d->p.dialog_authx509.retval = remmina_init_dialog_authx509( d->p.dialog_authx509.dialog, d->p.dialog_authx509.cacert,
92  d->p.dialog_authx509.cacrl, d->p.dialog_authx509.clientcert, d->p.dialog_authx509.clientkey );
93  break;
94  case FUNC_FTP_CLIENT_UPDATE_TASK:
95  remmina_ftp_client_update_task( d->p.ftp_client_update_task.client, d->p.ftp_client_update_task.task );
96  break;
97  case FUNC_FTP_CLIENT_GET_WAITING_TASK:
98  d->p.ftp_client_get_waiting_task.retval = remmina_ftp_client_get_waiting_task( d->p.ftp_client_get_waiting_task.client );
99  break;
100  case FUNC_PROTOCOLWIDGET_EMIT_SIGNAL:
101  remmina_protocol_widget_emit_signal(d->p.protocolwidget_emit_signal.gp, d->p.protocolwidget_emit_signal.signal_name);
102  break;
103  case FUNC_SFTP_CLIENT_CONFIRM_RESUME:
104 #ifdef HAVE_LIBSSH
105  d->p.sftp_client_confirm_resume.retval = remmina_sftp_client_confirm_resume( d->p.sftp_client_confirm_resume.client,
106  d->p.sftp_client_confirm_resume.path );
107 #endif
108  break;
109  case FUNC_VTE_TERMINAL_SET_ENCODING_AND_PTY:
110 #if defined (HAVE_LIBSSH) && defined (HAVE_LIBVTE)
111  remmina_plugin_ssh_vte_terminal_set_encoding_and_pty( d->p.vte_terminal_set_encoding_and_pty.terminal,
112  d->p.vte_terminal_set_encoding_and_pty.codeset,
113  d->p.vte_terminal_set_encoding_and_pty.master,
114  d->p.vte_terminal_set_encoding_and_pty.slave);
115 #endif
116  break;
117  }
118  pthread_mutex_lock(&d->pt_mutex);
119  d->complete = TRUE;
120  pthread_cond_signal(&d->pt_cond);
121  pthread_mutex_unlock(&d->pt_mutex);
122  }
123  else
124  {
125  /* thread has been cancelled, so we must free d memory here */
126  g_free(d);
127  }
128  return G_SOURCE_REMOVE;
129 }

137 void remmina_masterthread_exec_and_wait(RemminaMTExecData *d)
138 {
139  d->cancelled = FALSE;
140  d->complete = FALSE;
141  pthread_cleanup_push(remmina_masterthread_exec_cleanup_handler, (void *)d);
142  pthread_mutex_init(&d->pt_mutex, NULL);
143  pthread_cond_init(&d->pt_cond, NULL);
144  gdk_threads_add_idle((GSourceFunc)remmina_masterthread_exec_callback, (gpointer)d);
145  pthread_mutex_lock(&d->pt_mutex);
146  while (!d->complete)
147  pthread_cond_wait(&d->pt_cond, &d->pt_mutex);
148  pthread_cleanup_pop(0);
149  pthread_mutex_destroy(&d->pt_mutex);
150  pthread_cond_destroy(&d->pt_cond);
151 }
