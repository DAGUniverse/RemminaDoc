Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

62 void remmina_string_list_on_cell_edited(GtkCellRendererText *cell, const gchar *path_string, const gchar *new_text)
63 {
64  TRACE_CALL("remmina_string_list_on_cell_edited");
65  gchar *text;
66  gchar *error;
67  GtkTreePath *path = gtk_tree_path_new_from_string(path_string);
68  GtkTreeIter iter;
69 
70  gtk_tree_model_get_iter(GTK_TREE_MODEL(string_list->liststore_items), &iter, path);
71  /* Remove delimitors from the string */
72  text = remmina_public_str_replace(new_text, STRING_DELIMITOR, " ");
73  if (cell == string_list->cellrenderertext_item1)
74  {
75  gtk_list_store_set(string_list->liststore_items, &iter, COLUMN_DESCRIPTION, text, -1);
76  }
77  else
78  {
79  /* Check for validation only in second field */
80  if (string_list->priv->validation_func)
81  {
82  if (!((*string_list->priv->validation_func)(text, &error)))
83  {
84  gtk_label_set_text(string_list->label_status, error);
85  gtk_widget_show(GTK_WIDGET(string_list->label_status));
86  g_free(error);

89  {
90  gtk_widget_hide(GTK_WIDGET(string_list->label_status));
91  }
92  }
93  gtk_list_store_set(string_list->liststore_items, &iter, COLUMN_VALUE, text, -1);
94  }
95  gtk_tree_path_free(path);

183 /* Load a string list by splitting a string value */
184 void remmina_string_list_set_text(const gchar *text, const gboolean clear_data)
185 {
186  TRACE_CALL("remmina_string_list_set_text");
187  GtkTreeIter iter;
188  gchar **items;
189  gchar **values;
190  gint i;
191  /* Clear the data before to load new items */
192  if (clear_data)
193  gtk_list_store_clear(string_list->liststore_items);
194  /* Split the string and insert each snippet in the string list */
195  items = g_strsplit(text, STRING_DELIMITOR, -1);
196  for (i = 0; i < g_strv_length(items); i++)
197  {
198  values = g_strsplit(items[i], string_list->priv->fields_separator, -1);
199  gtk_list_store_append(string_list->liststore_items, &iter);
200  if (g_strv_length(values) > 1)
201  {
202  /* Two columns data */
203  gtk_list_store_set(string_list->liststore_items, &iter,
204  COLUMN_DESCRIPTION, values[0],
205  COLUMN_VALUE, values[1],
206  -1);
207  }
208  else
209  {
210  /* Single column data */
211  gtk_list_store_set(string_list->liststore_items, &iter,
212  COLUMN_VALUE, values[0],
213  -1);
214  }
215  g_strfreev(values);
216  }
217  g_strfreev(items);
218  remmina_string_list_update_buttons_state();
219 }

221 /* Get a string value representing the string list */
222 gchar* remmina_string_list_get_text(void)
223 {
224  TRACE_CALL("remmina_string_list_get_text");
225  GString *str;
226  GtkTreeIter iter;
227  gboolean first;
228  gboolean ret;
229  const gchar *item_description;
230  const gchar *item_value;
231 
232  str = g_string_new(NULL);
233  first = TRUE;
234  /* Cycle each GtkTreeIter in the ListStore */
235  ret = gtk_tree_model_get_iter_first(GTK_TREE_MODEL(string_list->liststore_items), &iter);
236  while (ret)
237  {
238  gtk_tree_model_get(GTK_TREE_MODEL(string_list->liststore_items), &iter,
239  COLUMN_DESCRIPTION, &item_description,
240  COLUMN_VALUE, &item_value,
241  -1);
242  if (!item_description)
243  item_description = "";
244  if (item_value && strlen(item_value) > 0)
245  {
246  /* Add a delimitor after the first element */
247  if (!first)
248  {
249  g_string_append(str, STRING_DELIMITOR);
250  }
251  else
252  {
253  first = FALSE;
254  }
255  /* Add the description for two columns list */
256  if (string_list->priv->two_columns)
257  {
258  g_string_append(str, item_description);
259  g_string_append(str, string_list->priv->fields_separator);
260  }
261  /* Add the element to the string */
262  g_string_append(str, item_value);
263  }
264  ret = gtk_tree_model_iter_next(GTK_TREE_MODEL(string_list->liststore_items), &iter);
265  }
266  return g_string_free(str, FALSE);
267 }
