Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

104 static gint remmina_on_command_line(GApplication *app, GApplicationCommandLine *cmdline)
105 {
106  TRACE_CALL("remmina_on_command_line");
107 
108  gint status = 0;
109  gboolean executed = FALSE;
110  GVariantDict *opts;
111  gchar *str;
112  gchar *protocol;
113  gchar *server;
114 
115  opts = g_application_command_line_get_options_dict(cmdline);
116 
117  if (g_variant_dict_lookup_value(opts, "quit", NULL))
118  {
119  remmina_exec_command(REMMINA_COMMAND_EXIT, NULL);
120  executed = TRUE;
121  status = 1;
122  }
123 
124  if (g_variant_dict_lookup_value(opts, "about", NULL))
125  {
126  remmina_exec_command(REMMINA_COMMAND_ABOUT, NULL);
127  executed = TRUE;
128  }
129 
130  if (g_variant_dict_lookup(opts, "connect", "^ay", &str))
131  {
132  remmina_exec_command(REMMINA_COMMAND_CONNECT, str);
133  g_free(str);
134  executed = TRUE;
135  }
136 
137  if (g_variant_dict_lookup(opts, "edit", "^ay", &str))
138  {
139  remmina_exec_command(REMMINA_COMMAND_EDIT, str);
140  g_free(str);
141  executed = TRUE;
142  }
143 
144  if (g_variant_dict_lookup_value(opts, "new", NULL))
145  {
146  if (!g_variant_dict_lookup(opts, "protocol", "&s", &protocol))
147  protocol = NULL;
148 
149  if (g_variant_dict_lookup(opts, "server", "&s", &server))
150  {
151  str = g_strdup_printf("%s,%s", protocol, server);
152  }
153  else
154  {
155  str = g_strdup(protocol);
156  }
157 
158  remmina_exec_command(REMMINA_COMMAND_NEW, str);
159  g_free(str);
160  executed = TRUE;
161  }
162 
163  if (g_variant_dict_lookup(opts, "pref", "&s", &str))
164  {
165  remmina_exec_command(REMMINA_COMMAND_PREF, str);
166  executed = TRUE;
167  }
168 
169  if (g_variant_dict_lookup(opts, "plugin", "&s", &str))
170  {
171  remmina_exec_command(REMMINA_COMMAND_PLUGIN, str);
172  executed = TRUE;
173  }
174 
175  if (g_variant_dict_lookup_value(opts, "icon", NULL))
176  {
177  remmina_exec_command(REMMINA_COMMAND_NONE, NULL);
178  executed = TRUE;
179  }
180 
181  if (g_variant_dict_lookup_value(opts, "version", NULL))
182  {
183  remmina_exec_command(REMMINA_COMMAND_VERSION, NULL);
184  executed = TRUE;
185  }
186 
187  if (g_variant_dict_lookup_value(opts, "full-version", NULL))
188  {
189  remmina_exec_command(REMMINA_COMMAND_FULL_VERSION, NULL);
190  executed = TRUE;
191  }
192 
193  if (!executed)
194  {
195  remmina_exec_command(REMMINA_COMMAND_MAIN, NULL);
196  }
197 
198  return status;
199 }

232 int main(int argc, char* argv[])
233 {
234  TRACE_CALL("main");
235  GtkApplication *app;
236  const gchar *app_id;
237  int status;
238 
239  gdk_set_allowed_backends("x11,broadway,quartz,wayland,mir");
240 
241  remmina_masterthread_exec_save_main_thread_id();
242 
243  bindtextdomain(GETTEXT_PACKAGE, REMMINA_RUNTIME_LOCALEDIR);
244  bind_textdomain_codeset(GETTEXT_PACKAGE, "UTF-8");
245  textdomain(GETTEXT_PACKAGE);
246 
247 #ifdef HAVE_LIBGCRYPT
248 # if GCRYPT_VERSION_NUMBER < 0x010600
249  gcry_error_t e;
250  if (!gcrypt_thread_initialized)
251  {
252  if ((e = gcry_control (GCRYCTL_SET_THREAD_CBS, &gcry_threads_pthread)) != GPG_ERR_NO_ERROR)
253  {
254  return (-1);
255  }
256  gcrypt_thread_initialized++;
257  }
258 #endif /* !GCRYPT_VERSION_NUMBER */
259  gcry_check_version (NULL);
260  gcry_control (GCRYCTL_DISABLE_SECMEM, 0);
261  gcry_control (GCRYCTL_INITIALIZATION_FINISHED, 0);
262 #endif /* !HAVE_LIBGCRYPT */
263 
264  app_id = g_application_id_is_valid(UNIQUE_APPNAME) ? UNIQUE_APPNAME : NULL;
265  app = gtk_application_new(app_id, G_APPLICATION_HANDLES_COMMAND_LINE);
266  g_signal_connect(app, "startup", G_CALLBACK(remmina_on_startup), NULL);
267  g_signal_connect(app, "command-line", G_CALLBACK(remmina_on_command_line), NULL);
268  g_signal_connect(app, "handle-local-options", G_CALLBACK(remmina_on_local_cmdline), NULL);
269 
270  g_application_add_main_option_entries(G_APPLICATION(app), remmina_options);
271 
272  g_application_set_inactivity_timeout(G_APPLICATION(app), 10000);
273  status = g_application_run(G_APPLICATION(app), argc, argv);
274  g_object_unref(app);
275 
276  return status;
277 }
