Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

102 static gboolean remmina_external_tools_launcher(const gchar* filename, const gchar* scriptname, const gchar* shortname)
103 {
104  TRACE_CALL("remmina_external_tools_launcher");
105  RemminaFile *remminafile;
106  const char *env_format = "%s=%s";
107  char *env;
108  size_t envstrlen;
109  gchar launcher[MAX_PATH_LEN];
110 
111  g_snprintf(launcher, MAX_PATH_LEN, "%s/launcher.sh", REMMINA_RUNTIME_EXTERNAL_TOOLS_DIR);
112 
113  remminafile = remmina_file_load(filename);
114  GHashTableIter iter;
115  const gchar *key, *value;
116  g_hash_table_iter_init(&iter, remminafile->settings);
117  while (g_hash_table_iter_next(&iter, (gpointer*) &key, (gpointer*) &value))
118  {
119  envstrlen = strlen(key) +strlen(value) + strlen(env_format) + 1;
120  env = (char *) malloc(envstrlen);
121  if (env == NULL)
122  {
123  return -1;
124  }
125 
126  int retval = snprintf(env, envstrlen, env_format, key,value);
127  if (retval > 0 && (size_t) retval <= envstrlen)
128  {
129  if (putenv(env) !=0)
130  {
131  /* If putenv fails, we must free the unused space */
132  free(env);
133  }
134  }
135  }
136  /* Adds the window title for the terminal window */
137  const char *term_title_key = "remmina_term_title";
138  const char *term_title_val_prefix = "Remmina external tool";
139  envstrlen = strlen(term_title_key) + strlen(term_title_val_prefix) + strlen(shortname) + 7;
140  env = (char *) malloc(envstrlen);
141  if (env != NULL)
142  {
143  if (snprintf(env, envstrlen, "%s=%s: %s", term_title_key, term_title_val_prefix, shortname) )
144  {
145  if (putenv(env) != 0)
146  {
147  /* If putenv fails, we must free the unused space */
148  free(env);
149  }
150  }
151  }
152 
153  const size_t cmdlen = strlen(launcher) +strlen(scriptname) + 2;
154  gchar *cmd = (gchar *)malloc(cmdlen);
155  g_snprintf(cmd, cmdlen, "%s %s", launcher, scriptname);
156  system(cmd);
157  free(cmd);
158 
159  remmina_file_free(remminafile);
160 
161  return TRUE;
162 }
