Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

73 static gboolean remmina_plugin_manager_register_plugin(RemminaPlugin *plugin)
74 {
75  TRACE_CALL("remmina_plugin_manager_register_plugin");
76  if (plugin->type == REMMINA_PLUGIN_TYPE_SECRET)
77  {
78  if (remmina_secret_plugin)
79  {
80  g_print("Remmina plugin %s (type=%s) bypassed.\n", plugin->name,
81  _(remmina_plugin_type_name[plugin->type]));
82  return FALSE;
83  }
84  remmina_secret_plugin = (RemminaSecretPlugin*) plugin;
85  }
86  g_ptr_array_add(remmina_plugin_table, plugin);
87  g_ptr_array_sort(remmina_plugin_table, (GCompareFunc) remmina_plugin_manager_compare_func);
88  /* g_print("Remmina plugin %s (type=%s) registered.\n", plugin->name, _(remmina_plugin_type_name[plugin->type])); */
89  return TRUE;
90 }

192 static void remmina_plugin_manager_load_plugin(const gchar *name)
193 {
194  TRACE_CALL("remmina_plugin_manager_load_plugin");
195  GModule *module;
196  RemminaPluginEntryFunc entry;
197 
198  module = g_module_open(name, G_MODULE_BIND_LAZY | G_MODULE_BIND_LOCAL);
199 
200  if (!module)
201  {
202  g_print("Failed to load plugin: %s.\n", name);
203  g_print("Error: %s\n", g_module_error());
204  return;
205  }
206 
207  if (!g_module_symbol(module, "remmina_plugin_entry", (gpointer*) &entry))
208  {
209  g_print("Failed to locate plugin entry: %s.\n", name);
210  return;
211  }
212 
213  if (!entry(&remmina_plugin_manager_service))
214  {
215  g_print("Plugin entry returned false: %s.\n", name);
216  return;
217  }
218 
219  /* We don't close the module because we will need it throughout the process lifetime */
220 }

222 void remmina_plugin_manager_init(void)
223 {
224  TRACE_CALL("remmina_plugin_manager_init");
225  GDir *dir;
226  const gchar *name, *ptr;
227  gchar *fullpath;
228 
229  remmina_plugin_table = g_ptr_array_new();
230 
231  if (!g_module_supported())
232  {
233  g_print("Dynamic loading of plugins is not supported in this platform!\n");
234  return;
235  }
236 
237  dir = g_dir_open(REMMINA_RUNTIME_PLUGINDIR, 0, NULL);
238  if (dir == NULL)
239  return;
240  while ((name = g_dir_read_name(dir)) != NULL)
241  {
242  if ((ptr = strrchr(name, '.')) == NULL)
243  continue;
244  ptr++;
245  if (g_strcmp0(ptr, G_MODULE_SUFFIX) != 0)
246  continue;
247  fullpath = g_strdup_printf(REMMINA_RUNTIME_PLUGINDIR "/%s", name);
248  remmina_plugin_manager_load_plugin(fullpath);
249  g_free(fullpath);
250  }
251  g_dir_close(dir);
252 }
