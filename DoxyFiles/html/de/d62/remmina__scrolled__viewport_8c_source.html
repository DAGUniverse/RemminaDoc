Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

64 static gboolean remmina_scrolled_viewport_motion_timeout(gpointer data)
65 {
66  TRACE_CALL("remmina_scrolled_viewport_motion_timeout");
67  RemminaScrolledViewport *gsv;
68  GtkWidget *child;
69  GdkDisplay *display;
70 #if GTK_CHECK_VERSION(3, 20, 0)
71  GdkSeat *seat;
72 #else
73  GdkDeviceManager *device_manager;
74 #endif
75  GdkDevice *pointer;
76  GdkScreen *screen;
77  GdkWindow *gsvwin;
78  gint x, y, mx, my, w, h, rootx, rooty;
79  GtkAdjustment *adj;
80  gdouble value;
81 
82  if (!REMMINA_IS_SCROLLED_VIEWPORT(data))
83  return FALSE;
84  if (!GTK_IS_BIN(data))
85  return FALSE;
86  gsv = REMMINA_SCROLLED_VIEWPORT(data);
87  if (!gsv->viewport_motion)
88  return FALSE;
89  child = gtk_bin_get_child(GTK_BIN(gsv));
90  if (!GTK_IS_VIEWPORT(child))
91  return FALSE;
92 
93  gsvwin = gtk_widget_get_window(GTK_WIDGET(gsv));
94  if (!gsv)
95  return FALSE;
96 
97  display = gdk_display_get_default();
98  if (!display)
99  return FALSE;
100 #if GTK_CHECK_VERSION(3, 20, 0)
101  seat = gdk_display_get_default_seat(display);
102  pointer = gdk_seat_get_pointer(seat);
103 #else
104  device_manager = gdk_display_get_device_manager (display);
105  pointer = gdk_device_manager_get_client_pointer (device_manager);
106 #endif
107  gdk_device_get_position(pointer, &screen, &x, &y);
108 
109  w = gdk_window_get_width(gsvwin) + 2; // Add 2px of black scroll border
110  h = gdk_window_get_height(gsvwin) + 2; // Add 2px of black scroll border
111 
112  gdk_window_get_root_origin(gsvwin, &rootx, &rooty );
113 
114  x -= rootx;
115  y -= rooty;
116 
117  mx = (x == 0 ? -1 : (x >= w - 1 ? 1 : 0));
118  my = (y == 0 ? -1 : (y >= h - 1 ? 1 : 0));
119  if (mx != 0)
120  {
121  gint step = MAX(10, MIN(remmina_pref.auto_scroll_step, w / 5));
122  adj = gtk_scrollable_get_hadjustment(GTK_SCROLLABLE(child));
123  value = gtk_adjustment_get_value(GTK_ADJUSTMENT(adj)) + (gdouble)(mx * step);
124  value = MAX(0, MIN(value, gtk_adjustment_get_upper(GTK_ADJUSTMENT(adj)) - (gdouble) w + 2.0));
125  gtk_adjustment_set_value(GTK_ADJUSTMENT(adj), value);
126  }
127  if (my != 0)
128  {
129  gint step = MAX(10, MIN(remmina_pref.auto_scroll_step, h / 5));
130  adj = gtk_scrollable_get_vadjustment(GTK_SCROLLABLE(child));
131  value = gtk_adjustment_get_value(GTK_ADJUSTMENT(adj)) + (gdouble)(my * step);
132  value = MAX(0, MIN(value, gtk_adjustment_get_upper(GTK_ADJUSTMENT(adj)) - (gdouble) h + 2.0));
133  gtk_adjustment_set_value(GTK_ADJUSTMENT(adj), value);
134  }
135  return TRUE;
136 }
