Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

60 static gboolean cb_closewidget(GtkWidget *widget, gpointer data)
61 {
62  TRACE_CALL(__func__);
63  /* The correct way to close a remmina_connection_window is to send
64  * it a "delete-event" signal. Simply destroying it will not close
65  * all network connections */
66  if (REMMINA_IS_CONNECTION_WINDOW(widget))
67  return remmina_connection_window_delete(REMMINA_CONNECTION_WINDOW(widget));
68  return TRUE;
69 }

100 void remmina_application_condexit(RemminaCondExitType why)
101 {
102  TRACE_CALL(__func__);
103 
104  /* Exit remmina only if there are no interesting windows left:
105  * no main window, no systray menu, no connection window.
106  * This function is usually called after a disconnection */
107 
108  switch (why) {
109  case REMMINA_CONDEXIT_ONDISCONNECT:
110  // A connection has disconnected, should we exit remmina ?
111  if (remmina_widget_pool_count() < 1 && !remmina_main_get_window() && !remmina_icon_is_available())
112  remmina_exec_exitremmina();
113  break;
114  case REMMINA_CONDEXIT_ONMAINWINDELETE:
115  // Main window has been deleted
116  if (remmina_widget_pool_count() < 1 && !remmina_icon_is_available())
117  remmina_exec_exitremmina();
118  break;
119  case REMMINA_CONDEXIT_ONQUIT:
120  // Quit command has been sent from main window or appindicator/systray menu
121  // quit means QUIT.
122  remmina_widget_pool_foreach(disable_remmina_connection_window_delete_confirm_cb, NULL);
123  remmina_exec_exitremmina();
124  break;
125  }
126 }

128 void remmina_exec_command(RemminaCommandType command, const gchar* data)
129 {
130  TRACE_CALL(__func__);
131  gchar* s1;
132  gchar* s2;
133  GtkWidget* widget;
134  GtkWindow* mainwindow;
135  GtkDialog* prefdialog;
136  RemminaEntryPlugin* plugin;
137 
138  switch (command) {
139  case REMMINA_COMMAND_MAIN:
140  mainwindow = remmina_main_get_window();
141  if (mainwindow) {
142  gtk_window_present(mainwindow);
143  gtk_window_deiconify(GTK_WINDOW(mainwindow));
144  }else {
145  widget = remmina_main_new();
146  gtk_widget_show(widget);
147  }
148  break;
149 
150  case REMMINA_COMMAND_PREF:
151  prefdialog = remmina_pref_dialog_get_dialog();
152  if (prefdialog) {
153  gtk_window_present(GTK_WINDOW(prefdialog));
154  gtk_window_deiconify(GTK_WINDOW(prefdialog));
155  }else {
156  /* Create a new preference dialog */
157  widget = GTK_WIDGET(remmina_pref_dialog_new(atoi(data), NULL));
158  gtk_widget_show(widget);
159  }
160  break;
161 
162  case REMMINA_COMMAND_NEW:
163  s1 = (data ? strchr(data, ',') : NULL);
164  if (s1) {
165  s1 = g_strdup(data);
166  s2 = strchr(s1, ',');
167  *s2++ = '\0';
168  widget = remmina_file_editor_new_full(s2, s1);
169  g_free(s1);
170  }else {
171  widget = remmina_file_editor_new_full(NULL, data);
172  }
173  gtk_widget_show(widget);
174  break;
175 
176  case REMMINA_COMMAND_CONNECT:
177  remmina_connection_window_open_from_filename(data);
178  break;
179 
180  case REMMINA_COMMAND_EDIT:
181  widget = remmina_file_editor_new_from_filename(data);
182  if (widget)
183  gtk_widget_show(widget);
184  break;
185 
186  case REMMINA_COMMAND_ABOUT:
187  remmina_about_open(NULL);
188  break;
189 
190  case REMMINA_COMMAND_VERSION:
191  mainwindow = remmina_main_get_window();
192  if (mainwindow) {
193  remmina_about_open(NULL);
194  }else {
195  g_print("%s %s %s (git %s)\n", g_get_application_name(), ISSNAP, VERSION, REMMINA_GIT_REVISION);
196  /* As we do not use the "handle-local-options" signal, we have to exit Remmina */
197  remmina_exec_command(REMMINA_COMMAND_EXIT, NULL);
198  }
199 
200  break;
201 
202  case REMMINA_COMMAND_FULL_VERSION:
203  mainwindow = remmina_main_get_window();
204  if (mainwindow) {
205  /* Show th widget with the list of plugins and versions */
206  remmina_plugin_manager_show(mainwindow);
207  }else {
208  g_print("%s %s %s (git %s)\n", g_get_application_name(), ISSNAP, VERSION, REMMINA_GIT_REVISION);
209 
210  remmina_plugin_manager_show_stdout();
211  remmina_exec_command(REMMINA_COMMAND_EXIT, NULL);
212  }
213 
214  break;
215 
216 
217  case REMMINA_COMMAND_PLUGIN:
218  plugin = (RemminaEntryPlugin*)remmina_plugin_manager_get_plugin(REMMINA_PLUGIN_TYPE_ENTRY, data);
219  if (plugin) {
220  plugin->entry_func();
221  }else {
222  widget = gtk_message_dialog_new(NULL, GTK_DIALOG_MODAL, GTK_MESSAGE_ERROR, GTK_BUTTONS_OK,
223  _("Plugin %s is not registered."), data);
224  g_signal_connect(G_OBJECT(widget), "response", G_CALLBACK(gtk_widget_destroy), NULL);
225  gtk_widget_show(widget);
226  remmina_widget_pool_register(widget);
227  }
228  break;
229 
230  case REMMINA_COMMAND_EXIT:
231  remmina_widget_pool_foreach(disable_remmina_connection_window_delete_confirm_cb, NULL);
232  remmina_exec_exitremmina();
233  break;
234 
235  default:
236  break;
237  }
238 }
