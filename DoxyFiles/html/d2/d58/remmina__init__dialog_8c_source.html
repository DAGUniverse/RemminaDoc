Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

155 void remmina_init_dialog_set_status(RemminaInitDialog *dialog, const gchar *status_format, ...)
156 {
157  TRACE_CALL("remmina_init_dialog_set_status");
158  /* This function can be called from a non main thread */
159 
160  va_list args;
161 
162  if (!dialog)
163  return;
164 
165  if (status_format)
166  {
167  if (dialog->status)
168  g_free(dialog->status);
169 
170  va_start(args, status_format);
171  dialog->status = g_strdup_vprintf(status_format, args);
172  va_end(args);
173 
174  if ( remmina_masterthread_exec_is_main_thread() )
175  {
176  gtk_label_set_text(GTK_LABEL(dialog->status_label), dialog->status);
177  }
178  else
179  {
180  RemminaMTExecData *d;
181  d = (RemminaMTExecData*)g_malloc( sizeof(RemminaMTExecData) );
182  d->func = FUNC_GTK_LABEL_SET_TEXT;
183  d->p.gtk_label_set_text.label = GTK_LABEL(dialog->status_label);
184  d->p.gtk_label_set_text.str = dialog->status;
185  remmina_masterthread_exec_and_wait(d);
186  g_free(d);
187  }
188  }
