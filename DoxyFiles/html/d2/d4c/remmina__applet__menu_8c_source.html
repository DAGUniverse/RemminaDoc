Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

154 void remmina_applet_menu_add_item(RemminaAppletMenu *menu, RemminaAppletMenuItem *menuitem)
155 {
156  TRACE_CALL("remmina_applet_menu_add_item");
157  GtkWidget *submenu;
158  GtkWidget *groupmenuitem;
159  GtkMenuItem *submenuitem;
160  gchar *s, *p1, *p2, *mstr;
161  GList *childs, *child;
162  gint position;
163 
164  submenu = GTK_WIDGET(menu);
165  s = g_strdup(menuitem->group);
166  p1 = s;
167  p2 = p1 ? strchr(p1, '/') : NULL;
168  if (p2)
169  *p2++ = '\0';
170  while (p1 && p1[0])
171  {
172  groupmenuitem = NULL;
173  childs = gtk_container_get_children(GTK_CONTAINER(submenu));
174  position = -1;
175  for (child = g_list_first(childs); child; child = g_list_next(child))
176  {
177  if (!GTK_IS_MENU_ITEM(child->data))
178  continue;
179  position++;
180  submenuitem = GTK_MENU_ITEM(child->data);
181  if (gtk_menu_item_get_submenu(submenuitem))
182  {
183  mstr = (gchar*) g_object_get_data(G_OBJECT(submenuitem), "group");
184  if (g_strcmp0(p1, mstr) == 0)
185  {
186  /* Found existing group menu */
187  submenu = gtk_menu_item_get_submenu(submenuitem);
188  groupmenuitem = GTK_WIDGET(submenuitem);
189  break;
190  }
191  else
192  {
193  /* Redo comparison ignoring case and respecting international
194  * collation, to set menu sort order */
195  if (strcoll(p1, mstr) < 0)
196  {
197  submenu = remmina_applet_menu_add_group(submenu, p1, position, menuitem,
198  &groupmenuitem);
199  break;
200  }
201  }
202  }
203  else
204  {
205  submenu = remmina_applet_menu_add_group(submenu, p1, position, menuitem, &groupmenuitem);
206  break;
207  }
208 
209  }
210 
211  if (!child)
212  {
213  submenu = remmina_applet_menu_add_group(submenu, p1, -1, menuitem, &groupmenuitem);
214  }
215  g_list_free(childs);
216  if (groupmenuitem && !menu->priv->hide_count)
217  {
218  remmina_applet_menu_increase_group_count(groupmenuitem);
219  }
220  p1 = p2;
221  p2 = p1 ? strchr(p1, '/') : NULL;
222  if (p2)
223  *p2++ = '\0';
224  }
225  g_free(s);
226 
227  childs = gtk_container_get_children(GTK_CONTAINER(submenu));
228  position = -1;
229  for (child = g_list_first(childs); child; child = g_list_next(child))
230  {
231  if (!GTK_IS_MENU_ITEM(child->data))
232  continue;
233  position++;
234  submenuitem = GTK_MENU_ITEM(child->data);
235  if (gtk_menu_item_get_submenu(submenuitem))
236  continue;
237  if (!REMMINA_IS_APPLET_MENU_ITEM(submenuitem))
238  continue;
239  if (strcoll(menuitem->name, REMMINA_APPLET_MENU_ITEM(submenuitem)->name) <= 0)
240  {
241  gtk_menu_shell_insert(GTK_MENU_SHELL(submenu), GTK_WIDGET(menuitem), position);
242  break;
243  }
244  }
245  if (!child)
246  {
247  gtk_menu_shell_append(GTK_MENU_SHELL(submenu), GTK_WIDGET(menuitem));
248  }
249  remmina_applet_menu_register_item(menu, menuitem);
250 }

269 void remmina_applet_menu_populate(RemminaAppletMenu *menu)
270 {
271  TRACE_CALL("remmina_applet_menu_populate");
272  GtkWidget *menuitem;
273  gchar filename[MAX_PATH_LEN];
274  GDir *dir;
275  const gchar *name;
276 
277  dir = g_dir_open(remmina_file_get_datadir(), 0, NULL);
278  if (dir != NULL)
279  {
280  /* Iterate all remote desktop profiles */
281  while ((name = g_dir_read_name(dir)) != NULL)
282  {
283  if (!g_str_has_suffix(name, ".remmina"))
284  continue;
285  g_snprintf(filename, sizeof(filename), "%s/%s", remmina_file_get_datadir(), name);
286 
287  menuitem = remmina_applet_menu_item_new(REMMINA_APPLET_MENU_ITEM_FILE, filename);
288  if (menuitem != NULL)
289  {
290  remmina_applet_menu_add_item(menu, REMMINA_APPLET_MENU_ITEM(menuitem));
291  gtk_widget_show(menuitem);
292  }
293  }
294  g_dir_close(dir);
295  }
