Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

196 static void remmina_rdp_settings_grid_load_layout(RemminaPluginRdpsetGrid* grid)
197 {
198  TRACE_CALL("remmina_rdp_settings_grid_load_layout");
199  gint i;
200  gchar* s;
201  GtkTreeIter iter;
202  RDP_KEYBOARD_LAYOUT* layouts;
203 
204  gtk_list_store_append(grid->keyboard_layout_store, &iter);
205  gtk_list_store_set(grid->keyboard_layout_store, &iter, 0, 0, 1, _("<Auto detect>"), -1);
206 
207  if (rdp_keyboard_layout == 0)
208  gtk_combo_box_set_active(GTK_COMBO_BOX(grid->keyboard_layout_combo), 0);
209 
210  gtk_label_set_text(GTK_LABEL(grid->keyboard_layout_label), "-");
211 
212  layouts = freerdp_keyboard_get_layouts(RDP_KEYBOARD_LAYOUT_TYPE_STANDARD | RDP_KEYBOARD_LAYOUT_TYPE_VARIANT);
213 
214  for (i = 0; layouts[i].code; i++)
215  {
216  s = g_strdup_printf("%08X - %s", layouts[i].code, layouts[i].name);
217  gtk_list_store_append(grid->keyboard_layout_store, &iter);
218  gtk_list_store_set(grid->keyboard_layout_store, &iter, 0, layouts[i].code, 1, s, -1);
219 
220  if (rdp_keyboard_layout == layouts[i].code)
221  gtk_combo_box_set_active(GTK_COMBO_BOX(grid->keyboard_layout_combo), i + 1);
222 
223  if (keyboard_layout == layouts[i].code)
224  gtk_label_set_text(GTK_LABEL(grid->keyboard_layout_label), s);
225 
226  g_free(s);
227  }
228 
229  free(layouts);
230 }

624 void remmina_rdp_settings_get_orientation_scale_prefs(int *desktopOrientation, int *desktopScaleFactor, int *deviceScaleFactor)
625 {
626  TRACE_CALL("remmina_rdp_settings_get_scaling_prefs");
627 
628  /* See https://msdn.microsoft.com/en-us/library/cc240510.aspx */
629 
630  int orientation, dpsf, desf;
631  gchar* s;
632 
633  *desktopOrientation = *desktopScaleFactor = *deviceScaleFactor = 0;
634 
635  s = remmina_plugin_service->pref_get_value("rdp_desktopOrientation");
636  orientation = s ? atoi(s) : 0;
637  g_free(s);
638  if (orientation != 90 && orientation != 180 && orientation != 270)
639  orientation = 0;
640  *desktopOrientation = orientation;
641 
642  s = remmina_plugin_service->pref_get_value("rdp_desktopScaleFactor");
643  dpsf = s ? atoi(s) : 0;
644  g_free(s);
645  if (dpsf < 100 || dpsf > 500)
646  return;
647 
648  s = remmina_plugin_service->pref_get_value("rdp_deviceScaleFactor");
649  desf = s ? atoi(s) : 0;
650  g_free(s);
651  if (desf != 100 && desf != 140 && desf != 180)
652  return;
653 
654  *desktopScaleFactor = dpsf;
655  *deviceScaleFactor = desf;
