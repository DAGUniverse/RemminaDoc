Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

354 void remmina_public_get_server_port(const gchar *server, gint defaultport, gchar **host, gint *port)
355 {
356  TRACE_CALL("remmina_public_get_server_port");
357  gchar *str, *ptr, *ptr2;
358 
359  str = g_strdup(server);
360 
361  if (str)
362  {
363  /* [server]:port format */
364  ptr = strchr(str, '[');
365  if (ptr)
366  {
367  ptr++;
368  ptr2 = strchr(ptr, ']');
369  if (ptr2)
370  {
371  *ptr2++ = '\0';
372  if (*ptr2 == ':')
373  defaultport = atoi(ptr2 + 1);
374  }
375  if (host)
376  *host = g_strdup(ptr);
377  if (port)
378  *port = defaultport;
379  g_free(str);
380  return;
381  }
382 
383  /* server:port format, IPv6 cannot use this format */
384  ptr = strchr(str, ':');
385  if (ptr)
386  {
387  ptr2 = strchr(ptr + 1, ':');
388  if (ptr2 == NULL)
389  {
390  *ptr++ = '\0';
391  defaultport = atoi(ptr);
392  }
393  /* More than one ':' means this is IPv6 address. Treat it as a whole address */
394  }
395  }
396 
397  if (host)
398  *host = str;
399  else
400  g_free(str);
401  if (port)
402  *port = defaultport;
403 }
