Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

90 static void remmina_nx_session_manager_on_response(GtkWidget *dialog, gint response_id, RemminaProtocolWidget *gp)
91 {
92  TRACE_CALL("remmina_nx_session_manager_on_response");
93  RemminaPluginNxData *gpdata = GET_PLUGIN_DATA(gp);
94  gint event_type;
95 
96  remmina_nx_session_manager_set_sensitive(gp, FALSE);
97  if (response_id <= 0)
98  {
99  event_type = REMMINA_NX_EVENT_CANCEL;
100  }
101  else
102  {
103  event_type = response_id;
104  }
105  if (response_id == REMMINA_NX_EVENT_TERMINATE && gpdata->manager_selected)
106  {
107  remmina_nx_session_iter_set(gpdata->nx, &gpdata->iter, REMMINA_NX_SESSION_COLUMN_STATUS, _("Terminating"));
108  }
109  if (response_id != REMMINA_NX_EVENT_TERMINATE)
110  {
111  gtk_widget_destroy(dialog);
112  gpdata->manager_dialog = NULL;
113  }
114  if (response_id != REMMINA_NX_EVENT_TERMINATE && response_id != REMMINA_NX_EVENT_CANCEL)
115  {
116  remmina_plugin_nx_service->protocol_plugin_init_show(gp);
117  }
118  remmina_nx_session_manager_send_signal(gpdata, event_type);
119 }
