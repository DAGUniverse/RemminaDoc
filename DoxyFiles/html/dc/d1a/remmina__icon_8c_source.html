Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

214 static void remmina_icon_populate_extra_menu_item(GtkWidget *menu)
215 {
216  TRACE_CALL(__func__);
217  GtkWidget *menuitem;
218  gboolean new_ontop;
219  GHashTableIter iter;
220  gchar *tmp;
221 
222  new_ontop = remmina_pref.applet_new_ontop;
223 
224  /* Iterate all discovered services from Avahi */
225  if (remmina_icon.avahi) {
226  g_hash_table_iter_init(&iter, remmina_icon.avahi->discovered_services);
227  while (g_hash_table_iter_next(&iter, NULL, (gpointer*)&tmp)) {
228  menuitem = remmina_applet_menu_item_new(REMMINA_APPLET_MENU_ITEM_DISCOVERED, tmp);
229  gtk_widget_show(menuitem);
230  remmina_applet_menu_add_item(REMMINA_APPLET_MENU(menu), REMMINA_APPLET_MENU_ITEM(menuitem));
231  }
232  }
233 
234  /* Separator */
235  menuitem = gtk_separator_menu_item_new();
236  gtk_widget_show(menuitem);
237  if (new_ontop) {
238  gtk_menu_shell_prepend(GTK_MENU_SHELL(menu), menuitem);
239  }else {
240  gtk_menu_shell_append(GTK_MENU_SHELL(menu), menuitem);
241  }
242 
243  /* New Connection */
244  menuitem = remmina_applet_menu_item_new(REMMINA_APPLET_MENU_ITEM_NEW);
245  gtk_widget_show(menuitem);
246  remmina_applet_menu_register_item(REMMINA_APPLET_MENU(menu), REMMINA_APPLET_MENU_ITEM(menuitem));
247  if (new_ontop) {
248  gtk_menu_shell_prepend(GTK_MENU_SHELL(menu), menuitem);
249  }else {
250  gtk_menu_shell_append(GTK_MENU_SHELL(menu), menuitem);
251  }
252 
253  g_signal_connect(G_OBJECT(menu), "launch-item", G_CALLBACK(remmina_icon_on_launch_item), NULL);
254  g_signal_connect(G_OBJECT(menu), "edit-item", G_CALLBACK(remmina_icon_on_edit_item), NULL);
255 }

362 gboolean remmina_icon_is_available(void)
363 {
364  /* Return TRUE if a remmina_icon (status indicator/systray menu) is
365  * available and shown to the user, so the user can continue
366  * its work without the remmina main window */
367 
368  TRACE_CALL(__func__);
369  gchar *gsversion;
370  unsigned int gsv_maj, gsv_min, gsv_seq;
371  gboolean gshell_has_legacyTray;
372 
373  if (!remmina_icon.icon)
374  return FALSE;
375  if (remmina_pref.disable_tray_icon)
376  return FALSE;
377 
378  /* Special treatmen under Gnome Shell */
379  if ((gsversion = remmina_sysinfo_get_gnome_shell_version()) != NULL) {
380  if (sscanf(gsversion, "%u.%u", &gsv_maj, &gsv_min) == 2)
381  gsv_seq = gsv_maj << 16 | gsv_min << 8;
382  else
383  gsv_seq = 0x030000;
384  g_free(gsversion);
385 
386  gshell_has_legacyTray = FALSE;
387  if (gsv_seq >= 0x031000 && gsv_seq <= 0x031800) {
388  /* Gnome shell from 3.16 to 3.24, Status Icon (GtkStatusIcon) is visible on the drawer
389  * at the bottom left of the screen */
390  gshell_has_legacyTray = TRUE;
391  }
392 
393 
394 #ifdef HAVE_LIBAPPINDICATOR
395  /* Gnome Shell with compiled in LIBAPPINDICATOR:
396  * ensure have also a working appindicator extension available */
397  if (remmina_sysinfo_is_appindicator_available()) {
398  /* No libappindicator extension for gnome shell, no remmina_icon */
399  return TRUE;
400  } else if (gshell_has_legacyTray) {
401  return TRUE;
402  } else {
403  return FALSE;
404  }
405 #endif
406  /* Gnome Shell without LIBAPPINDICATOR */
407  if (gshell_has_legacyTray) {
408  return TRUE;
409  }else {
410  /* Gnome shell < 3.16, Status Icon (GtkStatusIcon) is hidden
411  * on the message tray */
412  return FALSE;
413  }
414  }
415  return TRUE;
416 }

473  g_signal_connect(G_OBJECT(remmina_icon.icon), "activate", G_CALLBACK(remmina_icon_on_activate), NULL);
474 #endif
