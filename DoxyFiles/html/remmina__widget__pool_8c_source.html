Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

95 
96  for (i = 0; i < remmina_widget_pool->len; i++)
97  {
98  widget = GTK_WIDGET(g_ptr_array_index(remmina_widget_pool, i));
99  if (!G_TYPE_CHECK_INSTANCE_TYPE(widget, type))
100  continue;
101  /* gdk_window_get_toplevel won't work here, if the window is an embedded client. So we iterate the window tree */
102  for (parent = window; parent && parent != GDK_WINDOW_ROOT; parent = gdk_window_get_parent(parent))
103  {
104  if (gtk_widget_get_window(widget) == parent)
105  return widget;
106  }
107  }
108  return NULL;

111 gint remmina_widget_pool_foreach(RemminaWidgetPoolForEachFunc callback, gpointer data)
112 {
113  TRACE_CALL("remmina_widget_pool_foreach");
114  GtkWidget *widget;
115  gint i;
116  gint n = 0;
117  GPtrArray *wpcpy = NULL;
118 
119  if (remmina_widget_pool == NULL)
120  return 0;
121 
122  /* Make a copy of remmina_widget_pool, so we can survive when callback()
123  * remove an element from remmina_widget_pool */
124 
125  wpcpy = g_ptr_array_sized_new(remmina_widget_pool->len);
126 
127  for (i = 0; i < remmina_widget_pool->len; i++)
128  g_ptr_array_add(wpcpy, g_ptr_array_index(remmina_widget_pool, i));
129 
130  /* Scan the remmina_widget_pool and call callbac() on every element */
131  for (i = 0; i < wpcpy->len; i++)
132  {
133  widget = GTK_WIDGET(g_ptr_array_index(wpcpy, i));
134  if (callback(widget, data))
135  n++;
136  }
137 
138  /* Free the copy */
139  g_ptr_array_unref(wpcpy);
140  return n;
141 }
