Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

67 static gpointer
68 remmina_plugin_sftp_main_thread (gpointer data)
69 {
70  TRACE_CALL("remmina_plugin_sftp_main_thread");
71  RemminaProtocolWidget *gp = (RemminaProtocolWidget*) data;
72  RemminaPluginSftpData *gpdata;
73  RemminaFile *remminafile;
74  RemminaSSH *ssh;
75  RemminaSFTP *sftp = NULL;
76  gboolean cont = FALSE;
77  gint ret;
78  const gchar *cs;
79 
80  pthread_setcancelstate (PTHREAD_CANCEL_ENABLE, NULL);
81  CANCEL_ASYNC
82 
83  gpdata = GET_PLUGIN_DATA(gp);
84 
85  ssh = g_object_get_data (G_OBJECT(gp), "user-data");
86  if (ssh)
87  {
88  /* Create SFTP connection based on existing SSH session */
89  sftp = remmina_sftp_new_from_ssh (ssh);
90  if (remmina_ssh_init_session (REMMINA_SSH (sftp)) &&
91  remmina_ssh_auth (REMMINA_SSH (sftp), NULL) > 0 &&
92  remmina_sftp_open (sftp))
93  {
94  cont = TRUE;
95  }
96  }
97  else
98  {
99  /* New SFTP connection */
100  remminafile = remmina_plugin_service->protocol_plugin_get_file (gp);
101  remmina_plugin_service->file_set_string (remminafile, "ssh_server",
102  remmina_plugin_service->file_get_string (remminafile, "server"));
103 
104  sftp = remmina_sftp_new_from_file (remminafile);
105  while (1)
106  {
107  if (!remmina_ssh_init_session (REMMINA_SSH (sftp)))
108  {
109  remmina_plugin_service->protocol_plugin_set_error (gp, "%s", REMMINA_SSH (sftp)->error);
110  break;
111  }
112 
113  ret = remmina_ssh_auth_gui (REMMINA_SSH (sftp),
114  REMMINA_INIT_DIALOG (remmina_protocol_widget_get_init_dialog (gp)), remminafile);
115  if (ret == 0)
116  {
117  remmina_plugin_service->protocol_plugin_set_error (gp, "%s", REMMINA_SSH (sftp)->error);
118  }
119  if (ret <= 0) break;
120 
121  if (!remmina_sftp_open (sftp))
122  {
123  remmina_plugin_service->protocol_plugin_set_error (gp, "%s", REMMINA_SSH (sftp)->error);
124  break;
125  }
126 
127  cs = remmina_plugin_service->file_get_string (remminafile, "execpath");
128  if (cs && cs[0])
129  {
130  remmina_ftp_client_set_dir (REMMINA_FTP_CLIENT (gpdata->client), cs);
131  }
132 
133  cont = TRUE;
134  break;
135  }
136  }
137  if (!cont)
138  {
139  if (sftp) remmina_sftp_free (sftp);
140  IDLE_ADD ((GSourceFunc) remmina_plugin_service->protocol_plugin_close_connection, gp);
141  return NULL;
142  }
143 
144  remmina_sftp_client_open (REMMINA_SFTP_CLIENT (gpdata->client), sftp);
145  /* RemminaSFTPClient owns the object, we just take the reference */
146  gpdata->sftp = sftp;
147 
148  remmina_plugin_service->protocol_plugin_emit_signal (gp, "connect");
149 
150  gpdata->thread = 0;
151  return NULL;
152 }

218 remmina_plugin_sftp_close_connection (RemminaProtocolWidget *gp)
219 {
220  TRACE_CALL("remmina_plugin_sftp_close_connection");
221  RemminaPluginSftpData *gpdata = GET_PLUGIN_DATA(gp);
222  RemminaFile *remminafile;
223 
224  remminafile = remmina_plugin_service->protocol_plugin_get_file (gp);
225  if (gpdata->thread)
226  {
227  pthread_cancel (gpdata->thread);
228  if (gpdata->thread) pthread_join (gpdata->thread, NULL);
229  }
230 
231  remmina_ftp_client_save_state (REMMINA_FTP_CLIENT (gpdata->client), remminafile);
232  remmina_plugin_service->protocol_plugin_emit_signal (gp, "disconnect");
233  /* The session preference overwrite_all is always saved to FALSE in order
234  * to avoid unwanted overwriting.
235  * If we'd change idea just remove the next line to save the preference. */
236  remmina_file_set_int(remminafile,
237  REMMINA_PLUGIN_SFTP_FEATURE_PREF_OVERWRITE_ALL_KEY, FALSE);
238  return FALSE;

248 static void
249 remmina_plugin_sftp_call_feature (RemminaProtocolWidget *gp, const RemminaProtocolFeature *feature)
250 {
251  TRACE_CALL("remmina_plugin_sftp_call_feature");
252  RemminaPluginSftpData *gpdata = GET_PLUGIN_DATA(gp);
253  RemminaFile *remminafile;
254 
255  remminafile = remmina_plugin_service->protocol_plugin_get_file (gp);
256  switch (feature->id)
257  {
258  case REMMINA_PROTOCOL_FEATURE_TOOL_SSH:
259  remmina_plugin_service->open_connection (
260  remmina_file_dup_temp_protocol (remmina_plugin_service->protocol_plugin_get_file (gp), "SSH"),
261  NULL, gpdata->sftp, NULL);
262  return;
263  case REMMINA_PROTOCOL_FEATURE_TOOL_SFTP:
264  remmina_plugin_service->open_connection (
265  remmina_file_dup_temp_protocol (remmina_plugin_service->protocol_plugin_get_file (gp), "SFTP"),
266  NULL, gpdata->sftp, NULL);
267  return;
268  case REMMINA_PLUGIN_SFTP_FEATURE_PREF_SHOW_HIDDEN:
269  remmina_ftp_client_set_show_hidden (REMMINA_FTP_CLIENT (gpdata->client),
270  remmina_plugin_service->file_get_int (remminafile, "showhidden", FALSE));
271  return;
272  case REMMINA_PLUGIN_SFTP_FEATURE_PREF_OVERWRITE_ALL:
273  remmina_ftp_client_set_overwrite_status (REMMINA_FTP_CLIENT (gpdata->client),
274  remmina_plugin_service->file_get_int (remminafile,
275  REMMINA_PLUGIN_SFTP_FEATURE_PREF_OVERWRITE_ALL_KEY, FALSE));
276  return;
277  }
