Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

83 static gboolean remmina_plugin_xdmcp_start_xephyr(RemminaProtocolWidget *gp)
84 {
85  TRACE_CALL("remmina_plugin_xdmcp_start_xephyr");
86  RemminaPluginXdmcpData *gpdata = GET_PLUGIN_DATA(gp);
87  RemminaFile *remminafile;
88  gchar *argv[50];
89  gint argc;
90  gchar *host;
91  gint i;
92  GError *error = NULL;
93  gboolean ret;
94 
95  remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
96 
97  gpdata->display = remmina_get_available_xdisplay();
98  if (gpdata->display == 0)
99  {
100  remmina_plugin_service->protocol_plugin_set_error(gp, _("Run out of available local X display number."));
101  return FALSE;
102  }
103 
104  argc = 0;
105  argv[argc++] = g_strdup("Xephyr");
106 
107  argv[argc++] = g_strdup_printf(":%i", gpdata->display);
108 
109  argv[argc++] = g_strdup("-parent");
110  argv[argc++] = g_strdup_printf("%i", gpdata->socket_id);
111 
112  /* All Xephyr version between 1.5.0 and 1.6.4 will break when -screen argument is specified with -parent.
113  * It's not possible to support color depth if you have those Xephyr version. Please see this bug
114  * http://bugs.freedesktop.org/show_bug.cgi?id=24144
115  * As a workaround, a "Default" color depth will not add the -screen argument.
116  */
117  i = remmina_plugin_service->file_get_int(remminafile, "colordepth", 8);
118  if (i >= 8)
119  {
120  argv[argc++] = g_strdup("-screen");
121  argv[argc++] = g_strdup_printf("%ix%ix%i",
122  remmina_plugin_service->file_get_int(remminafile, "resolution_width", 640),
123  remmina_plugin_service->file_get_int(remminafile, "resolution_height", 480), i);
124  }
125 
126  if (i == 2)
127  {
128  argv[argc++] = g_strdup("-grayscale");
129  }
130 
131  if (remmina_plugin_service->file_get_int(remminafile, "showcursor", FALSE))
132  {
133  argv[argc++] = g_strdup("-host-cursor");
134  }
135  if (remmina_plugin_service->file_get_int(remminafile, "once", FALSE))
136  {
137  argv[argc++] = g_strdup("-once");
138  }
139  /* Listen on protocol TCP */
140  if (remmina_plugin_service->file_get_int(remminafile, "listen_on_tcp", FALSE))
141  {
142  argv[argc++] = g_strdup("-listen");
143  argv[argc++] = g_strdup("tcp");
144  }
145 
146  if (!remmina_plugin_service->file_get_int(remminafile, "ssh_enabled", FALSE))
147  {
148  remmina_plugin_service->get_server_port(remmina_plugin_service->file_get_string(remminafile, "server"), 0,
149  &host, &i);
150 
151  argv[argc++] = g_strdup("-query");
152  argv[argc++] = host;
153 
154  if (i)
155  {
156  argv[argc++] = g_strdup("-port");
157  argv[argc++] = g_strdup_printf("%i", i);
158  }
159  }
160  else
161  {
162  /* When the connection is through an SSH tunnel, it connects back to local unix socket,
163  * so for security we can disable tcp listening */
164  argv[argc++] = g_strdup("-nolisten");
165  argv[argc++] = g_strdup("tcp");
166 
167  /* FIXME: It's better to get the magic cookie back from xqproxy, then call xauth,
168  * instead of disable access control */
169  argv[argc++] = g_strdup("-ac");
170  }
171 
172  argv[argc++] = NULL;
173 
174  ret = g_spawn_async(NULL, argv, NULL, G_SPAWN_SEARCH_PATH, NULL, NULL, &gpdata->pid, &error);
175  for (i = 0; i < argc; i++)
176  g_free(argv[i]);
177 
178  if (!ret)
179  {
180  remmina_plugin_service->protocol_plugin_set_error(gp, "%s", error->message);
181  return FALSE;
182  }
183 
184  return TRUE;
185 }

187 static gboolean remmina_plugin_xdmcp_tunnel_init_callback(RemminaProtocolWidget *gp, gint remotedisplay, const gchar *server,
188  gint port)
189 {
190  TRACE_CALL("remmina_plugin_xdmcp_tunnel_init_callback");
191  RemminaPluginXdmcpData *gpdata = GET_PLUGIN_DATA(gp);
192  RemminaFile *remminafile;
193 
194  remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
195 
196  if (!remmina_plugin_xdmcp_start_xephyr(gp))
197  return FALSE;
198  while (!gpdata->ready)
199  sleep(1);
200 
201  remmina_plugin_service->protocol_plugin_set_display(gp, gpdata->display);
202 
203  if (remmina_plugin_service->file_get_string(remminafile, "exec"))
204  {
205  return remmina_plugin_service->protocol_plugin_ssh_exec(gp, FALSE, "DISPLAY=localhost:%i.0 %s", remotedisplay,
206  remmina_plugin_service->file_get_string(remminafile, "exec"));
207  }
208  else
209  {
210  return remmina_plugin_service->protocol_plugin_ssh_exec(gp, TRUE,
211  "xqproxy -display %i -host %s -port %i -query -manage", remotedisplay, server, port);
212  }
213 }

215 static gboolean remmina_plugin_xdmcp_main(RemminaProtocolWidget *gp)
216 {
217  TRACE_CALL("remmina_plugin_xdmcp_main");
218  RemminaPluginXdmcpData *gpdata = GET_PLUGIN_DATA(gp);
219  RemminaFile *remminafile;
220 
221  remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
222 
223  if (remmina_plugin_service->file_get_int(remminafile, "ssh_enabled", FALSE))
224  {
225  if (!remmina_plugin_service->protocol_plugin_start_xport_tunnel(gp, remmina_plugin_xdmcp_tunnel_init_callback))
226  {
227  gpdata->thread = 0;

229  }
230  }
231  else
232  {
233  if (!remmina_plugin_xdmcp_start_xephyr(gp))
234  {
235  gpdata->thread = 0;
236  return FALSE;
237  }
238  }
239 
240  gpdata->thread = 0;
241  return TRUE;
242 }

277 static gboolean remmina_plugin_xdmcp_open_connection(RemminaProtocolWidget *gp)
278 {
279  TRACE_CALL("remmina_plugin_xdmcp_open_connection");
280  RemminaPluginXdmcpData *gpdata = GET_PLUGIN_DATA(gp);
281  RemminaFile *remminafile;
282  gint width, height;
283 
284  if (!remmina_plugin_service->gtksocket_available())
285  {
286  remmina_plugin_service->protocol_plugin_set_error (gp,
287  _("Protocol %s is unavailable because GtkSocket only works under Xorg"),
288  remmina_plugin_xdmcp.name);
289  return FALSE;
290  }
291 
292  remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
293 
294  width = remmina_plugin_service->file_get_int(remminafile, "resolution_width", 640);
295  height = remmina_plugin_service->file_get_int(remminafile, "resolution_height", 480);
296  remmina_plugin_service->protocol_plugin_set_width(gp, width);
297  remmina_plugin_service->protocol_plugin_set_height(gp, height);
298  gtk_widget_set_size_request(GTK_WIDGET(gp), width, height);
299  gpdata->socket_id = gtk_socket_get_id(GTK_SOCKET(gpdata->socket));
300 
301 
302  if (remmina_plugin_service->file_get_int (remminafile, "ssh_enabled", FALSE))
303  {
304  if (pthread_create (&gpdata->thread, NULL, remmina_plugin_xdmcp_main_thread, gp))
305  {
306  remmina_plugin_service->protocol_plugin_set_error (gp,
307  "Failed to initialize pthread. Falling back to non-thread mode...");
308  gpdata->thread = 0;
309  return FALSE;
310  }
311  else
312  {
313  return TRUE;
314  }
315  }
316  else
317  {
318  return remmina_plugin_xdmcp_main (gp);
319  }

323 static gboolean remmina_plugin_xdmcp_close_connection(RemminaProtocolWidget *gp)
324 {
325  TRACE_CALL("remmina_plugin_xdmcp_close_connection");
326  RemminaPluginXdmcpData *gpdata = GET_PLUGIN_DATA(gp);
327 
328  if (gpdata->thread)
329  {
330  pthread_cancel (gpdata->thread);
331  if (gpdata->thread) pthread_join (gpdata->thread, NULL);
332  }
333 
334  if (gpdata->pid)
335  {
336  kill(gpdata->pid, SIGTERM);
337  g_spawn_close_pid(gpdata->pid);
338  gpdata->pid = 0;
339  }
340 
341  remmina_plugin_service->protocol_plugin_emit_signal(gp, "disconnect");
342 
343  return FALSE;
344 }
