Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

356 /* Send to the plugin some keystrokes */
357 void remmina_protocol_widget_send_keystrokes(RemminaProtocolWidget* gp, GtkMenuItem *widget)
358 {
359  TRACE_CALL(__func__);
360  gchar *keystrokes = g_object_get_data(G_OBJECT(widget), "keystrokes");
361  guint *keyvals;
362  gint i;
363  GdkKeymap *keymap = gdk_keymap_get_default();
364  gchar *iter = keystrokes;
365  gunichar character;
366  guint keyval;
367  GdkKeymapKey *keys;
368  gint n_keys;
369  /* Single keystroke replace */
370  typedef struct _KeystrokeReplace {
371  gchar *search;
372  gchar *replace;
373  guint keyval;
374  } KeystrokeReplace;
375  /* Special characters to replace */
376  KeystrokeReplace keystrokes_replaces[] =
377  {
378  { "\\n", "\n", GDK_KEY_Return },
379  { "\\t", "\t", GDK_KEY_Tab },
380  { "\\b", "\b", GDK_KEY_BackSpace },
381  { "\\e", "\e", GDK_KEY_Escape },
382  { "\\\\", "\\", GDK_KEY_backslash },
383  { NULL, NULL, 0 }
384  };
385  /* Keystrokes can be sent only to plugins that accepts them */
386  if (remmina_protocol_widget_plugin_receives_keystrokes(gp)) {
387  /* Replace special characters */
388  for (i = 0; keystrokes_replaces[i].replace; i++) {
389  remmina_public_str_replace_in_place(keystrokes,
390  keystrokes_replaces[i].search,
391  keystrokes_replaces[i].replace);
392  }
393  keyvals = (guint*)g_malloc(strlen(keystrokes));
394  while (TRUE) {
395  /* Process each character in the keystrokes */
396  character = g_utf8_get_char_validated(iter, -1);
397  if (character == 0)
398  break;
399  keyval = gdk_unicode_to_keyval(character);
400  /* Replace all the special character with its keyval */
401  for (i = 0; keystrokes_replaces[i].replace; i++) {
402  if (character == keystrokes_replaces[i].replace[0]) {
403  keys = g_new0(GdkKeymapKey, 1);
404  keyval = keystrokes_replaces[i].keyval;
405  /* A special character was generated, no keyval lookup needed */
406  character = 0;
407  break;
408  }
409  }
410  /* Decode character if it's not a special character */
411  if (character) {
412  /* get keyval without modifications */
413  if (!gdk_keymap_get_entries_for_keyval(keymap, keyval, &keys, &n_keys)) {
414  g_warning("keyval 0x%04x has no keycode!", keyval);
415  iter = g_utf8_find_next_char(iter, NULL);
416  continue;
417  }
418  }
419  /* Add modifier keys */
420  n_keys = 0;
421  if (keys->level & 1)
422  keyvals[n_keys++] = GDK_KEY_Shift_L;
423  if (keys->level & 2)
424  keyvals[n_keys++] = GDK_KEY_Alt_R;
425  keyvals[n_keys++] = keyval;
426  /* Send keystroke to the plugin */
427  gp->priv->plugin->send_keystrokes(gp, keyvals, n_keys);
428  g_free(keys);
429  /* Process next character in the keystrokes */
430  iter = g_utf8_find_next_char(iter, NULL);
431  }
432  g_free(keyvals);
433  }
434  g_free(keystrokes);
435  return;
436 }

744  ret = TRUE;
745  }
746  }else {
747  remmina_ssh_set_error(REMMINA_SSH(tunnel), _("Failed to execute command: %s"));
