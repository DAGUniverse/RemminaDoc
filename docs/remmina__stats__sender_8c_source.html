Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

112 static gchar *rsa_encrypt_string(RSA *pubKey, const char *instr)
113 {
114  TRACE_CALL(__func__);
115  /* Calls RSA_public_encrypt multiple times to encrypt instr.
116  * At the end, base64 encode the resulting buffer
117  * Return a buffer ptr. Use g_free() to deallocate it */
118 
119  int rsaLen = RSA_size(pubKey);
120  int inLen = strlen(instr);
121  int remaining, r;
122  int blksz, maxblksz;
123  int ebufSize;
124  unsigned char *ebuf, *outptr;
125  gchar *enc;
126 
127  maxblksz = rsaLen - 12;
128  ebufSize = (((inLen - 1) / maxblksz) + 1) * rsaLen;
129  ebuf = g_malloc(ebufSize);
130  outptr = ebuf;
131  remaining = strlen(instr);
132 
133  while(remaining > 0) {
134  blksz = remaining > maxblksz ? maxblksz : remaining;
135  r = RSA_public_encrypt(blksz,
136  (const unsigned char *)instr,
137  outptr,
138  pubKey, RSA_PKCS1_PADDING); /* Our poor JS libraries only supports RSA_PKCS1_PADDING */
139  if (r == -1 ) {
140  unsigned long e;
141  ERR_load_crypto_strings();
142  e = ERR_get_error();
143  g_print("Error RSA_public_encrypt(): %s - func: %s - reason: %s\n", ERR_lib_error_string(e), ERR_func_error_string(e), ERR_reason_error_string(e));
144  g_free(ebuf);
145  ERR_free_strings();
146  return NULL;
147  }
148  instr += blksz;
149  remaining -= blksz;
150  outptr += r;
151  }
152 
153  enc = g_base64_encode(ebuf, ebufSize);
154  g_free(ebuf);
155 
156  return enc;
157

159 }

161 static gboolean remmina_stats_collector_done(gpointer data)
162 {
163  TRACE_CALL(__func__);
164  JsonNode *n;
165  JsonGenerator *g;
166  gchar *unenc_s, *enc_s;
167  SoupSession *ss;
168  SoupMessage *msg;
169  JsonBuilder *b;
170  JsonObject *o;
171  BIO *pkbio;
172  RSA *pubkey;
173  gchar *uid;
174  sc_tdata *sctdata;
175 
176  sctdata = (sc_tdata *)data;
177  if (sctdata == NULL)
178  return G_SOURCE_REMOVE;
179 
180  n = sctdata->statsroot;
181  if (n == NULL) {
182  g_free(data);
183  return G_SOURCE_REMOVE;
184  }
185 
186  if ((o = json_node_get_object(n)) == NULL) {
187  g_free(data);
188  return G_SOURCE_REMOVE;
189  }
190 
191  uid = g_strdup(json_object_get_string_member(o, "UID"));
192 
193  g = json_generator_new();
194  json_generator_set_root(g, n);
195  json_node_unref(n);
196  unenc_s = json_generator_to_data(g, NULL); // unenc_s=serialized stats
197  remmina_log_printf("STATS upload: JSON data%s\n", unenc_s);
198  g_object_unref(g);
199 
200  /* Now encrypt "s" with remminastats public key */
201 
202  pkbio = BIO_new_mem_buf(remmina_RSA_PubKey_v1, -1);
203  pubkey = PEM_read_bio_RSA_PUBKEY(pkbio, NULL, NULL, NULL);
204  if (pubkey == NULL) {
205  ERR_load_crypto_strings();
206  unsigned long e;
207  e = ERR_get_error();
208  g_print("Failure in PEM_read_bio_RSAPublicKey: %s - func: %s - reason: %s\n", ERR_lib_error_string(e), ERR_func_error_string(e), ERR_reason_error_string(e));
209  g_print("%s\n", ERR_error_string( e, NULL ));
210  BIO_free(pkbio);
211  g_free(unenc_s);
212  ERR_free_strings();
213  g_free(data);
214  return G_SOURCE_REMOVE;
215  }
216 
217  enc_s = rsa_encrypt_string(pubkey, unenc_s);
218 
219  g_free(unenc_s);
220  BIO_free(pkbio);
221 
222 
223  /* Create new json encrypted object */
224 
225  b = json_builder_new();
226  json_builder_begin_object(b);
227  json_builder_set_member_name(b, "keyversion");
228  json_builder_add_int_value(b, 1);
229  json_builder_set_member_name(b, "encdata");
230  json_builder_add_string_value(b, enc_s);
231  json_builder_set_member_name(b, "UID");
232  json_builder_add_string_value(b, uid);
233  json_builder_end_object(b);
234  n = json_builder_get_root(b);
235  g_object_unref(b);
236 
237  g_free(uid);
238  g_free(enc_s);
239 
240  if (!sctdata->show_only) {
241 
242  g = json_generator_new();
243  json_generator_set_root(g, n);
244  enc_s = json_generator_to_data(g, NULL); // unenc_s=serialized stats
245  g_object_unref(g);
246 
247  ss = soup_session_new();
248  msg = soup_message_new("POST", PERIODIC_UPLOAD_URL);
249  soup_message_set_request(msg, "application/json",
250  SOUP_MEMORY_COPY, enc_s, strlen(enc_s));
251  soup_session_queue_message(ss, msg, soup_callback, enc_s);
252 
253  remmina_log_printf("STATS upload: Starting upload to url %s\n", PERIODIC_UPLOAD_URL);
254  }
255 
256  json_node_unref(n);
257  g_free(data);
258 
259  return G_SOURCE_REMOVE;
260 }

301 static gboolean remmina_stats_sender_periodic_check(gpointer user_data)
302 {
303  TRACE_CALL(__func__);
304  GTimeVal t;
305  glong next;
306 
307  if (!remmina_stat_sender_can_send())
308  return G_SOURCE_REMOVE;
309 
310  /* Calculate "next" upload time based on last sent time */
311  next = remmina_pref.periodic_usage_stats_last_sent + PERIODIC_UPLOAD_INTERVAL_SEC;
312  g_get_current_time(&t);
313  /* If current time is after "next" or clock is going back (but > 1/1/2018), then do send stats */
314  if (t.tv_sec > next || (t.tv_sec < remmina_pref.periodic_usage_stats_last_sent && t.tv_sec > 1514764800)) {
315  remmina_stats_sender_send(FALSE);
316  }
317 
318  periodic_check_counter++;
319  if (periodic_check_counter <= 1) {
320  /* Reschedule periodic check less frequently after 1st tick.
321  * Note that PERIODIC_CHECK_INTERVAL_MS becomes also a retry interval in case of
322  * upload failure */
323  periodic_check_source = g_timeout_add_full(G_PRIORITY_LOW, PERIODIC_CHECK_INTERVAL_MS, remmina_stats_sender_periodic_check, NULL, NULL);
324  return G_SOURCE_REMOVE;
325  }
326  return G_SOURCE_CONTINUE;
327 }
