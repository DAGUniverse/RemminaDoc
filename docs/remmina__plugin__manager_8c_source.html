Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

78  g_print("Remmina plugin %s (type=%s) bypassed.\n", plugin->name,

191 static void remmina_plugin_manager_load_plugin(const gchar *name)
192 {
193  TRACE_CALL(__func__);
194  GModule *module;
195  RemminaPluginEntryFunc entry;
196 
197  module = g_module_open(name, G_MODULE_BIND_LAZY | G_MODULE_BIND_LOCAL);
198 
199  if (!module) {
200  g_print("Failed to load plugin: %s.\n", name);
201  g_print("Error: %s\n", g_module_error());
202  return;
203  }
204 
205  if (!g_module_symbol(module, "remmina_plugin_entry", (gpointer*)&entry)) {
206  g_print("Failed to locate plugin entry: %s.\n", name);
207  return;
208  }
209 
210  if (!entry(&remmina_plugin_manager_service)) {
211  g_print("Plugin entry returned false: %s.\n", name);
212  return;
213  }
214 
215  /* We don't close the module because we will need it throughout the process lifetime */
216 }

218 void remmina_plugin_manager_init(void)
219 {
220  TRACE_CALL(__func__);
221  GDir *dir;
222  const gchar *name, *ptr;
223  gchar *fullpath;
224 
225  remmina_plugin_table = g_ptr_array_new();
226 
227  if (!g_module_supported()) {
228  g_print("Dynamic loading of plugins is not supported in this platform!\n");
229  return;
230  }
231 
232  dir = g_dir_open(REMMINA_RUNTIME_PLUGINDIR, 0, NULL);
233  if (dir == NULL)
234  return;
235  while ((name = g_dir_read_name(dir)) != NULL) {
236  if ((ptr = strrchr(name, '.')) == NULL)
237  continue;
238  ptr++;
239  if (g_strcmp0(ptr, G_MODULE_SUFFIX) != 0)
240  continue;
241  fullpath = g_strdup_printf(REMMINA_RUNTIME_PLUGINDIR "/%s", name);
242  remmina_plugin_manager_load_plugin(fullpath);
243  g_free(fullpath);
244  }
245  g_dir_close(dir);
246 }
