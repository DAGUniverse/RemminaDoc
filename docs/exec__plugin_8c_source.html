Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

137 static gboolean remmina_plugin_exec_run(RemminaProtocolWidget *gp)
138 {
139  TRACE_CALL(__func__);
140  RemminaFile* remminafile;
141  const gchar *cmd;
142  gchar *stdout_buffer;
143  gchar *stderr_buffer;
144  char **argv;
145  GError *error = NULL;
146  GPid child_pid;
147  gint child_stdout, child_stderr;
148  GtkDialog *dialog;
149  GIOChannel *out_ch, *err_ch;
150 
151  remmina_plugin_service->log_printf("[%s] Plugin run\n", PLUGIN_NAME);
152  RemminaPluginExecData *gpdata = GET_PLUGIN_DATA(gp);
153  remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
154 
155  cmd = remmina_plugin_service->file_get_string(remminafile, "execcommand");
156  if (!cmd) {
157  gtk_text_buffer_set_text (gpdata->log_buffer,
158  _("You did not set any command to be executed"), -1);
159  remmina_plugin_service->protocol_plugin_emit_signal(gp, "connect");
160  return TRUE;
161  }
162 
163  g_shell_parse_argv(cmd, NULL, &argv, &error);
164  if (error) {
165  gtk_text_buffer_set_text (gpdata->log_buffer, error->message, -1);
166  remmina_plugin_service->protocol_plugin_emit_signal(gp, "connect");
167  return TRUE;
168  g_error_free(error);
169  }
170 
171  if (remmina_plugin_service->file_get_int(remminafile, "runasync", FALSE)) {
172  remmina_plugin_service->log_printf("[%s] Run Async\n", PLUGIN_NAME);
173  g_spawn_async_with_pipes( NULL,
174  argv,
175  NULL,
176  G_SPAWN_DO_NOT_REAP_CHILD |
177  G_SPAWN_SEARCH_PATH_FROM_ENVP |
178  G_SPAWN_SEARCH_PATH,
179  NULL,
180  NULL,
181  &child_pid,
182  NULL,
183  &child_stdout,
184  &child_stderr,
185  &error);
186  if (error != NULL) {
187  gtk_text_buffer_set_text (gpdata->log_buffer, error->message, -1);
188  g_error_free(error);
189  remmina_plugin_service->protocol_plugin_emit_signal(gp, "connect");
190  return TRUE;
191  }
192  g_child_watch_add(child_pid, (GChildWatchFunc)cb_child_watch, gp );
193 
194  /* Create channels that will be used to read data from pipes. */
195  out_ch = g_io_channel_unix_new(child_stdout);
196  err_ch = g_io_channel_unix_new(child_stderr);
197  /* Add watches to channels */
198  g_io_add_watch(out_ch, G_IO_IN | G_IO_HUP, (GIOFunc)cb_out_watch, gp );
199  g_io_add_watch(err_ch, G_IO_IN | G_IO_HUP, (GIOFunc)cb_err_watch, gp );
200 
201  }else {
202  dialog = GTK_DIALOG(gtk_message_dialog_new(NULL, GTK_DIALOG_MODAL,
203  GTK_MESSAGE_QUESTION, GTK_BUTTONS_YES_NO,
204  _("WARNING! Executing a command synchronously, may hung Remmina.\rDo you really want to continue?")));
205  gint result = gtk_dialog_run (GTK_DIALOG (dialog));
206 
207  switch (result)
208  {
209  case GTK_RESPONSE_YES:
210  break;
211  default:
212  gtk_widget_destroy(GTK_WIDGET(dialog));
213  return FALSE;
214  break;
215  }
216  gtk_widget_destroy(GTK_WIDGET(dialog));
217  remmina_plugin_service->log_printf("[%s] Run Sync\n", PLUGIN_NAME);
218  g_spawn_sync (NULL, // CWD or NULL
219  argv,
220  NULL, // ENVP or NULL
221  G_SPAWN_SEARCH_PATH |
222  G_SPAWN_SEARCH_PATH_FROM_ENVP,
223  NULL,
224  NULL,
225  &stdout_buffer, // STDOUT
226  &stderr_buffer, // STDERR
227  NULL, // Exit status
228  &error);
229  }
230  if (!error) {
231  remmina_plugin_service->log_printf("[%s] Command executed\n", PLUGIN_NAME);
232  gtk_text_buffer_set_text (gpdata->log_buffer, stdout_buffer, -1);
233  }else {
234  g_warning("Command %s exited with error: %s\n", cmd, error->message);
235  gtk_text_buffer_set_text (gpdata->log_buffer, error->message, -1);
236  g_error_free(error);
237  }
238 
239  g_strfreev(argv);
240 
241  remmina_plugin_service->protocol_plugin_emit_signal(gp, "connect");
242  return TRUE;
243 }
