Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

158  if (setting_iter) {

161  return setting_iter;

166  setting_iter = protocol_plugin->advanced_settings;

170  return setting_iter;
171  setting_iter++;
172  }
173  }
174 
175  return NULL;

205 RemminaFile*
206 remmina_file_load(const gchar *filename)
207 {
208  TRACE_CALL(__func__);
209  GKeyFile *gkeyfile;
210  RemminaFile *remminafile;
211  gchar *proto;
212  gchar **keys;
213  gchar *key;
214  gchar *resolution_str;
215  gint i;
216  gchar *s, *sec;
217  RemminaProtocolPlugin* protocol_plugin;
218  RemminaSecretPlugin *secret_plugin;
219  int w, h;
220 
221  gkeyfile = g_key_file_new();
222 
223  if (!g_key_file_load_from_file(gkeyfile, filename, G_KEY_FILE_NONE, NULL)) {
224  g_key_file_free(gkeyfile);
225  return NULL;
226  }
227 
228  if (g_key_file_has_key(gkeyfile, "remmina", "name", NULL)) {
229  remminafile = remmina_file_new_empty();
230 
231  protocol_plugin = NULL;
232 
233  /* Identify the protocol plugin and get pointers to its RemminaProtocolSetting structs */
234  proto = g_key_file_get_string(gkeyfile, "remmina", "protocol", NULL);
235  if (proto) {
236  protocol_plugin = (RemminaProtocolPlugin*)remmina_plugin_manager_get_plugin(REMMINA_PLUGIN_TYPE_PROTOCOL, proto);
237  g_free(proto);
238  }
239 
240  secret_plugin = remmina_plugin_manager_get_secret_plugin();
241  remminafile->filename = g_strdup(filename);
242  keys = g_key_file_get_keys(gkeyfile, "remmina", NULL, NULL);
243  if (keys) {
244  for (i = 0; keys[i]; i++) {
245  key = keys[i];
246  if (is_encrypted_setting_by_name(key, protocol_plugin)) {
247  s = g_key_file_get_string(gkeyfile, "remmina", key, NULL);
248  if (g_strcmp0(s, ".") == 0) {
249  if (secret_plugin) {
250  sec = secret_plugin->get_password(remminafile, key);
251  remmina_file_set_string(remminafile, key, sec);
252  /* Annotate in spsettings that this value comes from secret_plugin */
253  g_hash_table_insert(remminafile->spsettings, g_strdup(key), NULL);
254  g_free(sec);
255  }else {
256  remmina_file_set_string(remminafile, key, s);
257  }
258  }else {
259  remmina_file_set_string_ref(remminafile, key, remmina_crypt_decrypt(s));
260  }
261  g_free(s);
262  }else {
263  /* If we find "resolution", then we split it in two */
264  if (strcmp(key, "resolution") == 0) {
265  resolution_str = g_key_file_get_string(gkeyfile, "remmina", key, NULL);
266  if (remmina_public_split_resolution_string(resolution_str, &w, &h)) {
267  remmina_file_set_string_ref(remminafile, "resolution_width", g_strdup_printf("%i", w));
268  remmina_file_set_string_ref(remminafile, "resolution_height", g_strdup_printf("%i", h));
269  } else {
270  remmina_file_set_string_ref(remminafile, "resolution_width", NULL);
271  remmina_file_set_string_ref(remminafile, "resolution_height", NULL);
272  }
273  g_free(resolution_str);
274  }else {
275  remmina_file_set_string_ref(remminafile, key,
276  g_key_file_get_string(gkeyfile, "remmina", key, NULL));
277  }
278  }
279  }
280  g_strfreev(keys);
281  }
282  }else {
283  remminafile = NULL;
284  }
285 
286  g_key_file_free(gkeyfile);
287 
288  return remminafile;
289 }

316 const gchar*
317 remmina_file_get_string(RemminaFile *remminafile, const gchar *setting)
318 {
319  TRACE_CALL(__func__);
320  gchar *value;
321  const gchar* message;
322 
323  /* Returned value is a pointer to the string stored on the hash table,
324  * please do not free it or the hash table will contain invalid pointer */
325  if ( !remmina_masterthread_exec_is_main_thread() ) {
326  /* Allow the execution of this function from a non main thread
327  * (plugins needs it to have user credentials)*/
328  RemminaMTExecData *d;
329  const gchar *retval;
330  d = (RemminaMTExecData*)g_malloc( sizeof(RemminaMTExecData) );
331  d->func = FUNC_FILE_GET_STRING;
332  d->p.file_get_string.remminafile = remminafile;
333  d->p.file_get_string.setting = setting;
334  remmina_masterthread_exec_and_wait(d);
335  retval = d->p.file_get_string.retval;
336  g_free(d);
337  return retval;
338  }
339 
340  if (strcmp(setting, "resolution") == 0) {
341  message = "WARNING: the \"resolution\" setting in .pref files is deprecated, but some code in remmina or in a plugin is trying to read it.\n";
342  fputs(message, stdout);
343  remmina_main_show_warning_dialog(message);
344  return NULL;
345  }
346 
347  value = (gchar*)g_hash_table_lookup(remminafile->settings, setting);
348  return value && value[0] ? value : NULL;
349 }

405 void remmina_file_save(RemminaFile *remminafile)
406 {
407  TRACE_CALL(__func__);
408  RemminaSecretPlugin *secret_plugin;
409  RemminaProtocolPlugin* protocol_plugin;
410  GHashTableIter iter;
411  const gchar *key, *value;
412  gchar *s, *proto, *content;
413  GKeyFile *gkeyfile;
414  gsize length = 0;
415 
416  if (remminafile->prevent_saving)
417  return;
418 
419  if ((gkeyfile = remmina_file_get_keyfile(remminafile)) == NULL)
420  return;
421 
422  /* Identify the protocol plugin and get pointers to its RemminaProtocolSetting structs */
423  proto = (gchar*)g_hash_table_lookup(remminafile->settings, "protocol");
424  if (proto) {
425  protocol_plugin = (RemminaProtocolPlugin*)remmina_plugin_manager_get_plugin(REMMINA_PLUGIN_TYPE_PROTOCOL, proto);
426  } else {
427  printf("Remmina WARNING: saving settings for unknown protocol, because remminafile has non proto key\n");
428  protocol_plugin = NULL;
429  }
430 
431  secret_plugin = remmina_plugin_manager_get_secret_plugin();
432 
433  g_hash_table_iter_init(&iter, remminafile->settings);
434  while (g_hash_table_iter_next(&iter, (gpointer*)&key, (gpointer*)&value)) {
435  if (is_encrypted_setting_by_name(key, protocol_plugin)) {
436  if (remminafile->filename && g_strcmp0(remminafile->filename, remmina_pref_file)) {
437  if (secret_plugin) {
438  if (value && value[0]) {
439  if (g_strcmp0(value, ".") != 0) {
440  secret_plugin->store_password(remminafile, key, value);
441  }
442  g_key_file_set_string(gkeyfile, "remmina", key, ".");
443  }else {
444  g_key_file_set_string(gkeyfile, "remmina", key, "");
445  secret_plugin->delete_password(remminafile, key);
446  }
447  }else {
448  if (value && value[0]) {
449  s = remmina_crypt_encrypt(value);
450  g_key_file_set_string(gkeyfile, "remmina", key, s);
451  g_free(s);
452  }else {
453  g_key_file_set_string(gkeyfile, "remmina", key, "");
454  }
455  }
456  }
457  }else {
458  g_key_file_set_string(gkeyfile, "remmina", key, value);
459  }
460  }
461 
462  /* Avoid storing redundant and deprecated "resolution" field */
463  g_key_file_remove_key(gkeyfile, "remmina", "resolution", NULL);
464 
465  /* Store gkeyfile to disk (password are already sent to keyring) */
466  content = g_key_file_to_data(gkeyfile, &length, NULL);
467  g_file_set_contents(remminafile->filename, content, length, NULL);
468 
469  g_free(content);
470  g_key_file_free(gkeyfile);
471 
472  remmina_main_update_file_datetime(remminafile);
473 }

552 void remmina_file_unsave_password(RemminaFile *remminafile)
553 {

555 
556  TRACE_CALL(__func__);
557  const RemminaProtocolSetting* setting_iter;
558  RemminaProtocolPlugin* protocol_plugin;
559  gchar *proto;
560  protocol_plugin = NULL;
561 
562  remmina_file_set_string(remminafile, "password", NULL);
563 
564  proto = (gchar*)g_hash_table_lookup(remminafile->settings, "protocol");
565  if (proto) {

578  setting_iter = protocol_plugin->advanced_settings;
579  if (setting_iter) {
580  while (setting_iter->type != REMMINA_PROTOCOL_SETTING_TYPE_END) {
581  if (is_encrypted_setting(setting_iter)) {
582  remmina_file_set_string(remminafile, remmina_plugin_manager_get_canonical_setting_name(setting_iter), NULL);
583  }
584  setting_iter++;
585  }
586  }

591 gchar*
592 remmina_file_get_datetime(RemminaFile *remminafile)
593 {
594  TRACE_CALL(__func__);
595 
596  GFile *file;
597  GFileInfo *info;
598 
599  struct timeval tv;
600  struct tm* ptm;
601  char time_string[256];
602 
603  guint64 mtime;
604  gchar *modtime_string;
605 
606  file = g_file_new_for_path(remminafile->filename);
607 
608  info = g_file_query_info(file,
609  G_FILE_ATTRIBUTE_TIME_MODIFIED,
610  G_FILE_QUERY_INFO_NONE,
611  NULL,
612  NULL);
613 
614  g_object_unref(file);
615 
616  if (info == NULL) {
617  g_print("couldn't get time info\n");
618  return "26/01/1976 23:30:00";
619  }
620 
621  mtime = g_file_info_get_attribute_uint64(info, G_FILE_ATTRIBUTE_TIME_MODIFIED);
622  tv.tv_sec = mtime;
623 
624  ptm = localtime(&tv.tv_sec);
