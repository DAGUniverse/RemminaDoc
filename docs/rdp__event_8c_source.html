Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

173 static void remmina_rdp_event_scale_area(RemminaProtocolWidget* gp, gint* x, gint* y, gint* w, gint* h)
174 {
175  TRACE_CALL(__func__);
176  gint width, height;
177  gint sx, sy, sw, sh;
178  rfContext* rfi = GET_PLUGIN_DATA(gp);
179 
180  if (!rfi || !rfi->connected || rfi->is_reconnecting || !rfi->surface)
181  return;
182 
183  width = remmina_plugin_service->protocol_plugin_get_width(gp);
184  height = remmina_plugin_service->protocol_plugin_get_height(gp);
185 
186  if ((width == 0) || (height == 0))
187  return;
188 
189  if ((rfi->scale_width == width) && (rfi->scale_height == height)) {
190  /* Same size, just copy the pixels */
191  *x = MIN(MAX(0, *x), width - 1);
192  *y = MIN(MAX(0, *y), height - 1);
193  *w = MIN(width - *x, *w);
194  *h = MIN(height - *y, *h);
195  return;
196  }
197 
198  /* We have to extend the scaled region one scaled pixel, to avoid gaps */
199 
200  sx = MIN(MAX(0, (*x) * rfi->scale_width / width
201  - rfi->scale_width / width - 2), rfi->scale_width - 1);
202 
203  sy = MIN(MAX(0, (*y) * rfi->scale_height / height
204  - rfi->scale_height / height - 2), rfi->scale_height - 1);
205 
206  sw = MIN(rfi->scale_width - sx, (*w) * rfi->scale_width / width
207  + rfi->scale_width / width + 4);
208 
209  sh = MIN(rfi->scale_height - sy, (*h) * rfi->scale_height / height
210  + rfi->scale_height / height + 4);
211 
212  *x = sx;
213  *y = sy;
214  *w = sw;
215  *h = sh;
216 }

577  if (!rfi || !rfi->connected || rfi->is_reconnecting)
578  return FALSE;
579 
580 #ifdef ENABLE_GTK_INSPECTOR_KEY
581  /* GTK inspector key is propagated up. Disabled by default.
582  * enable it by defining ENABLE_GTK_INSPECTOR_KEY */
583  if ( ( event->state & GDK_CONTROL_MASK ) != 0 && ( event->keyval == GDK_KEY_I || event->keyval == GDK_KEY_D ) ) {
584  return FALSE;
585  }
586 #endif
587 
588  rdp_event.type = REMMINA_RDP_EVENT_TYPE_SCANCODE;
589  rdp_event.key_event.up = (event->type == GDK_KEY_PRESS ? False : True);
590  rdp_event.key_event.extended = False;
591 
592  switch (event->keyval) {
593  case GDK_KEY_Pause:
594  /*
595  * See https://msdn.microsoft.com/en-us/library/cc240584.aspx
596  * 2.2.8.1.1.3.1.1.1 Keyboard Event (TS_KEYBOARD_EVENT)
597  * for pause key management
598  */
599  rdp_event.key_event.key_code = 0x1D;
600  rdp_event.key_event.up = False;
601  remmina_rdp_event_event_push(gp, &rdp_event);
602  rdp_event.key_event.key_code = 0x45;
603  rdp_event.key_event.up = False;
604  remmina_rdp_event_event_push(gp, &rdp_event);
605  rdp_event.key_event.key_code = 0x1D;
606  rdp_event.key_event.up = True;
607  remmina_rdp_event_event_push(gp, &rdp_event);
608  rdp_event.key_event.key_code = 0x45;
609  rdp_event.key_event.up = True;
610  remmina_rdp_event_event_push(gp, &rdp_event);
611  break;
612 
613  default:
614  if (!rfi->use_client_keymap) {
615  scancode = freerdp_keyboard_get_rdp_scancode_from_x11_keycode(event->hardware_keycode);
616  rdp_event.key_event.key_code = scancode & 0xFF;
617  rdp_event.key_event.extended = scancode & 0x100;
618  if (rdp_event.key_event.key_code) {
619  remmina_rdp_event_event_push(gp, &rdp_event);
620  keypress_list_add(gp, rdp_event);
621  }
622  } else {
623  display = gtk_widget_get_display(widget);
624  unicode_keyval = gdk_keyval_to_unicode(event->keyval);
625  /* Decide when whe should send a keycode or a unicode character.
626  * - All non char keys (shift, alt, win) should be sent as keycode
627  * - Space should be sent as keycode (see issue #1364)
628  * - All special keys (F1-F10, numeric pad, home/end/arrows/pgup/pgdn/ins/del) keycode
629  * - All key pressed while CTRL or ALT or WIN is down are not decoded by gdk_keyval_to_unicode(), so send it as keycode
630  * - All keycodes not translatable to unicode chars, as keycode
631  * - The rest as unicode char
632  */
633  if (event->keyval >= 0xfe00 || // arrows, shift, alt, Fn, num keypad...
634  event->hardware_keycode == 0x41 || // space bar
635  unicode_keyval == 0 || // impossible to translate
636  (event->state & (GDK_MOD1_MASK | GDK_CONTROL_MASK | GDK_SUPER_MASK)) != 0 // a modifier not recognized by gdk_keyval_to_unicode()
637  ) {
638  scancode = freerdp_keyboard_get_rdp_scancode_from_x11_keycode(event->hardware_keycode);
639  rdp_event.key_event.key_code = scancode & 0xFF;
640  rdp_event.key_event.extended = scancode & 0x100;
641  if (rdp_event.key_event.key_code) {
642  remmina_rdp_event_event_push(gp, &rdp_event);
643  keypress_list_add(gp, rdp_event);
644  }
645  } else {
646  rdp_event.type = REMMINA_RDP_EVENT_TYPE_SCANCODE_UNICODE;
647  rdp_event.key_event.unicode_code = unicode_keyval;
648  rdp_event.key_event.extended = False;
649  remmina_rdp_event_event_push(gp, &rdp_event);
650  keypress_list_add(gp, rdp_event);
651  }
652  }
653  break;
654  }
655 
656  return TRUE;
