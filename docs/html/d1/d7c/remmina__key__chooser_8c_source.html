Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

44 static gboolean remmina_key_chooser_dialog_on_key_press(GtkWidget *widget, GdkEventKey *event, RemminaKeyChooserArguments *arguments)
45 {
46  TRACE_CALL(__func__);

48  arguments->state = event->state;

50  gtk_dialog_response(GTK_DIALOG(gtk_widget_get_toplevel(widget)),

56 /* Show a key chooser dialog and return the keyval for the selected key */
57 RemminaKeyChooserArguments* remmina_key_chooser_new(GtkWindow *parent_window, gboolean use_modifiers)
58 {
59  TRACE_CALL(__func__);
60  GtkBuilder *builder = remmina_public_gtk_builder_new_from_file("remmina_key_chooser.glade");
61  GtkDialog *dialog;
62  RemminaKeyChooserArguments *arguments;
63  arguments = g_new0(RemminaKeyChooserArguments, 1);
64  arguments->state = 0;
65  arguments->use_modifiers = use_modifiers;
66 
67  /* Setup the dialog */
68  dialog = GTK_DIALOG(gtk_builder_get_object(builder, "KeyChooserDialog"));
69  gtk_window_set_transient_for(GTK_WINDOW(dialog), parent_window);
70  /* Connect the GtkEventBox signal */
71  g_signal_connect(gtk_builder_get_object(builder, "eventbox_key_chooser"), "key-press-event",
72  G_CALLBACK(remmina_key_chooser_dialog_on_key_press), arguments);
73  /* Show the dialog and destroy it after the use */
74  arguments->response = gtk_dialog_run(dialog);
75  gtk_widget_destroy(GTK_WIDGET(dialog));
76  /* The delete button set the keyval 0 */
77  if (arguments->response == GTK_RESPONSE_REJECT)
78  arguments->keyval = 0;
79  return arguments;
80 }

83 gchar* remmina_key_chooser_get_value(guint keyval, guint state)
84 {
85  TRACE_CALL(__func__);
86 
87  if (!keyval)

89

91  state & GDK_SHIFT_MASK ? KEY_MODIFIER_SHIFT : "",

93  state & GDK_MOD1_MASK ? KEY_MODIFIER_ALT : "",
94  state & GDK_SUPER_MASK ? KEY_MODIFIER_SUPER : "",
95  state & GDK_HYPER_MASK ? KEY_MODIFIER_HYPER : "",
96  state & GDK_META_MASK ? KEY_MODIFIER_META : "",
97  gdk_keyval_name(gdk_keyval_to_upper(keyval)));
98 }

100 /* Get the keyval of a (lowercase) character value */
101 guint remmina_key_chooser_get_keyval(const gchar *value)
102 {
103  TRACE_CALL(__func__);
104  gchar *patterns[] =
105  {
106  KEY_MODIFIER_SHIFT,
107  KEY_MODIFIER_CTRL,
108  KEY_MODIFIER_ALT,
109  KEY_MODIFIER_SUPER,
110  KEY_MODIFIER_HYPER,
111  KEY_MODIFIER_META,
112  NULL
113  };
114  gint i;
115  gchar *tmpvalue;
116  gchar *newvalue;
117  guint keyval;
118 
119  if (g_strcmp0(value, KEY_CHOOSER_NONE) == 0)
120  return 0;
121 
122  /* Remove any modifier text before to get the keyval */
123  newvalue = g_strdup(value);
124  for (i = 0; i < g_strv_length(patterns); i++) {
125  tmpvalue = remmina_public_str_replace(newvalue, patterns[i], "");
126  g_free(newvalue);
127  newvalue = g_strdup(tmpvalue);
128  g_free(tmpvalue);
129  }
130  keyval = gdk_keyval_to_lower(gdk_keyval_from_name(newvalue));
131  g_free(newvalue);
132  return keyval;
133 }
