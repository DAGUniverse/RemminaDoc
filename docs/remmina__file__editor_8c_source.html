Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

673 static void remmina_file_editor_create_settings(RemminaFileEditor* gfe, GtkWidget* grid,
674  const RemminaProtocolSetting* settings)
675 {
676  TRACE_CALL(__func__);
677  RemminaFileEditorPriv* priv = gfe->priv;
678  GtkWidget* widget;
679  gint grid_row = 0;
680  gint grid_column = 0;
681  gchar** strarr;
682  gchar* setting_name;
683 
684 
685  while (settings->type != REMMINA_PROTOCOL_SETTING_TYPE_END) {
686  setting_name = (gchar*)(remmina_plugin_manager_get_canonical_setting_name(settings));
687  switch (settings->type) {
688  case REMMINA_PROTOCOL_SETTING_TYPE_SERVER:
689  remmina_file_editor_create_server(gfe, settings, grid, grid_row);
690  break;
691 
692  case REMMINA_PROTOCOL_SETTING_TYPE_PASSWORD:
693  widget = remmina_file_editor_create_password(gfe, grid, grid_row, 0,
694  g_dgettext(priv->plugin->domain, settings->label),
695  remmina_file_get_string(priv->remmina_file, setting_name));
696  g_hash_table_insert(priv->setting_widgets, setting_name, widget);
697  grid_row++;
698  break;
699 
700  case REMMINA_PROTOCOL_SETTING_TYPE_RESOLUTION:
701  remmina_file_editor_create_resolution(gfe, settings, grid, grid_row);
702  grid_row++;
703  break;
704 
705  case REMMINA_PROTOCOL_SETTING_TYPE_KEYMAP:
706  strarr = remmina_pref_keymap_groups();
707  priv->keymap_combo = remmina_file_editor_create_select(gfe, grid,
708  grid_row + 1, 0,
709  _("Keyboard mapping"), (const gpointer*)strarr,
710  remmina_file_get_string(priv->remmina_file, "keymap"));
711  g_strfreev(strarr);
712  grid_row++;
713  break;
714 
715  case REMMINA_PROTOCOL_SETTING_TYPE_TEXT:
716  widget = remmina_file_editor_create_text(gfe, grid, grid_row, 0,
717  g_dgettext(priv->plugin->domain, settings->label),
718  remmina_file_get_string(priv->remmina_file, setting_name));
719  g_hash_table_insert(priv->setting_widgets, setting_name, widget);
720  grid_row++;
721  break;
722 
723  case REMMINA_PROTOCOL_SETTING_TYPE_SELECT:
724  widget = remmina_file_editor_create_select(gfe, grid, grid_row, 0,
725  g_dgettext(priv->plugin->domain, settings->label),
726  (const gpointer*)settings->opt1,
727  remmina_file_get_string(priv->remmina_file, setting_name));
728  g_hash_table_insert(priv->setting_widgets, setting_name, widget);
729  break;
730 
731  case REMMINA_PROTOCOL_SETTING_TYPE_COMBO:
732  widget = remmina_file_editor_create_combo(gfe, grid, grid_row, 0,
733  g_dgettext(priv->plugin->domain, settings->label),
734  (const gchar*)settings->opt1,
735  remmina_file_get_string(priv->remmina_file, setting_name));
736  g_hash_table_insert(priv->setting_widgets, setting_name, widget);
737  break;
738 
739  case REMMINA_PROTOCOL_SETTING_TYPE_CHECK:
740  widget = remmina_file_editor_create_check(gfe, grid, grid_row, grid_column,
741  g_dgettext(priv->plugin->domain, settings->label),
742  remmina_file_get_int(priv->remmina_file, setting_name, FALSE));
743  g_hash_table_insert(priv->setting_widgets, setting_name, widget);
744  break;
745 
746  case REMMINA_PROTOCOL_SETTING_TYPE_FILE:
747  widget = remmina_file_editor_create_chooser(gfe, grid, grid_row, 0,
748  g_dgettext(priv->plugin->domain, settings->label),
749  remmina_file_get_string(priv->remmina_file, setting_name),
750  GTK_FILE_CHOOSER_ACTION_OPEN);
751  g_hash_table_insert(priv->setting_widgets, setting_name, widget);
752  break;
753 
754  case REMMINA_PROTOCOL_SETTING_TYPE_FOLDER:
755  widget = remmina_file_editor_create_chooser(gfe, grid, grid_row, 0,
756  g_dgettext(priv->plugin->domain, settings->label),
757  remmina_file_get_string(priv->remmina_file, setting_name),
758  GTK_FILE_CHOOSER_ACTION_SELECT_FOLDER);
759  g_hash_table_insert(priv->setting_widgets, setting_name, widget);
760  break;
761 
762  default:
763  break;
764  }
765  /* If the setting wants compactness, move to the next column */
766  if (settings->compact) {
767  grid_column++;
768  }
769  /* Add a new settings row and move to the first column
770  * if the setting doesn't want the compactness
771  * or we already have two columns */
772  if (!settings->compact || grid_column > 1) {
773  grid_row++;
774  grid_column = 0;
775  }
776  settings++;
777  }
