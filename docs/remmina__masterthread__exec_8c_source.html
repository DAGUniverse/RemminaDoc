Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

46 static gboolean remmina_masterthread_exec_callback(RemminaMTExecData *d)
47 {
48 
49  /* This function is called on main GTK Thread via gdk_threads_add_idlde()
50  * from remmina_masterthread_exec_and_wait() */
51 
52  if (!d->cancelled) {
53  switch (d->func) {
54  case FUNC_INIT_SAVE_CRED:
55  remmina_protocol_widget_init_save_cred(d->p.init_save_creds.gp);
56  break;
57  case FUNC_CHAT_RECEIVE:
58  remmina_protocol_widget_chat_receive(d->p.chat_receive.gp, d->p.chat_receive.text);
59  break;
60  case FUNC_FILE_GET_STRING:
61  d->p.file_get_string.retval = remmina_file_get_string( d->p.file_get_string.remminafile, d->p.file_get_string.setting );
62  break;
63  case FUNC_DIALOG_SERVERKEY_CONFIRM:
64  d->p.dialog_serverkey_confirm.retval = remmina_init_dialog_serverkey_confirm( d->p.dialog_serverkey_confirm.dialog,
65  d->p.dialog_serverkey_confirm.serverkey, d->p.dialog_serverkey_confirm.prompt );
66  break;
67  case FUNC_DIALOG_AUTHPWD:
68  d->p.dialog_authpwd.retval = remmina_init_dialog_authpwd(d->p.dialog_authpwd.dialog,
69  d->p.dialog_authpwd.label, d->p.dialog_authpwd.allow_save);
70  break;
71  case FUNC_GTK_LABEL_SET_TEXT:
72  gtk_label_set_text( d->p.gtk_label_set_text.label, d->p.gtk_label_set_text.str );
73  break;
74  case FUNC_DIALOG_AUTHUSERPWD:
75  d->p.dialog_authuserpwd.retval = remmina_init_dialog_authuserpwd( d->p.dialog_authuserpwd.dialog,
76  d->p.dialog_authuserpwd.want_domain, d->p.dialog_authuserpwd.default_username,
77  d->p.dialog_authuserpwd.default_domain, d->p.dialog_authuserpwd.allow_save );
78  break;
79  case FUNC_DIALOG_CERT:
80  d->p.dialog_certificate.retval = remmina_init_dialog_certificate( d->p.dialog_certificate.dialog,
81  d->p.dialog_certificate.subject, d->p.dialog_certificate.issuer, d->p.dialog_certificate.fingerprint );
82  break;
83  case FUNC_DIALOG_CERTCHANGED:
84  d->p.dialog_certchanged.retval = remmina_init_dialog_certificate_changed( d->p.dialog_certchanged.dialog,
85  d->p.dialog_certchanged.subject, d->p.dialog_certchanged.issuer, d->p.dialog_certchanged.new_fingerprint,
86  d->p.dialog_certchanged.old_fingerprint );
87  break;
88  case FUNC_DIALOG_AUTHX509:
89  d->p.dialog_authx509.retval = remmina_init_dialog_authx509( d->p.dialog_authx509.dialog, d->p.dialog_authx509.cacert,
90  d->p.dialog_authx509.cacrl, d->p.dialog_authx509.clientcert, d->p.dialog_authx509.clientkey );
91  break;
92  case FUNC_FTP_CLIENT_UPDATE_TASK:
93  remmina_ftp_client_update_task( d->p.ftp_client_update_task.client, d->p.ftp_client_update_task.task );
94  break;
95  case FUNC_FTP_CLIENT_GET_WAITING_TASK:
96  d->p.ftp_client_get_waiting_task.retval = remmina_ftp_client_get_waiting_task( d->p.ftp_client_get_waiting_task.client );
97  break;
98  case FUNC_PROTOCOLWIDGET_EMIT_SIGNAL:
99  remmina_protocol_widget_emit_signal(d->p.protocolwidget_emit_signal.gp, d->p.protocolwidget_emit_signal.signal_name);
100  break;
101  case FUNC_SFTP_CLIENT_CONFIRM_RESUME:
102 #ifdef HAVE_LIBSSH
103  d->p.sftp_client_confirm_resume.retval = remmina_sftp_client_confirm_resume( d->p.sftp_client_confirm_resume.client,
104  d->p.sftp_client_confirm_resume.path );
105 #endif
106  break;
107  case FUNC_VTE_TERMINAL_SET_ENCODING_AND_PTY:
108 #if defined (HAVE_LIBSSH) && defined (HAVE_LIBVTE)
109  remmina_plugin_ssh_vte_terminal_set_encoding_and_pty( d->p.vte_terminal_set_encoding_and_pty.terminal,
110  d->p.vte_terminal_set_encoding_and_pty.codeset,
111  d->p.vte_terminal_set_encoding_and_pty.master,
112  d->p.vte_terminal_set_encoding_and_pty.slave);
113 #endif
114  break;
115  }
116  pthread_mutex_lock(&d->pt_mutex);
117  d->complete = TRUE;
118  pthread_cond_signal(&d->pt_cond);
119  pthread_mutex_unlock(&d->pt_mutex);
120  }else {
121  /* thread has been cancelled, so we must free d memory here */
122  g_free(d);
123  }
124  return G_SOURCE_REMOVE;
125 }

134 void remmina_masterthread_exec_and_wait(RemminaMTExecData *d)
135 {
136  d->cancelled = FALSE;
137  d->complete = FALSE;
138  pthread_cleanup_push(remmina_masterthread_exec_cleanup_handler, (void*)d);
139  pthread_mutex_init(&d->pt_mutex, NULL);
140  pthread_cond_init(&d->pt_cond, NULL);
141  gdk_threads_add_idle((GSourceFunc)remmina_masterthread_exec_callback, (gpointer)d);
142  pthread_mutex_lock(&d->pt_mutex);
143  while (!d->complete)
144  pthread_cond_wait(&d->pt_cond, &d->pt_mutex);
145  pthread_cleanup_pop(0);
146  pthread_mutex_destroy(&d->pt_mutex);
147  pthread_cond_destroy(&d->pt_cond);
148 }
