Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

94 static gboolean remmina_plugin_spice_open_connection(RemminaProtocolWidget *gp)
95 {
96  TRACE_CALL(__func__);
97 
98  gint port;
99  const gchar *cacert;
100  gchar *host;
101  RemminaPluginSpiceData *gpdata = GET_PLUGIN_DATA(gp);
102  RemminaFile *remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
103 
104  remmina_plugin_service->get_server_port(remmina_plugin_service->file_get_string(remminafile, "server"),
105  XSPICE_DEFAULT_PORT,
106  &host,
107  &port);
108 
109  g_object_set(gpdata->session, "host", host, NULL);
110  g_free(host);
111 
112  /* Unencrypted connection */
113  if (!remmina_plugin_service->file_get_int(remminafile, "usetls", FALSE)) {
114  g_object_set(gpdata->session, "port", g_strdup_printf("%i", port), NULL);
115  }
116  /* TLS encrypted connection */
117  else{
118  g_object_set(gpdata->session, "tls_port", g_strdup_printf("%i", port), NULL);
119 
120  /* Server CA certificate */
121  cacert = remmina_plugin_service->file_get_string(remminafile, "cacert");
122  if (cacert) {
123  g_object_set(gpdata->session, "ca-file", cacert, NULL);
124  }
125  }
126 
127  spice_session_connect(gpdata->session);
128 
129  /*
130  * FIXME: Add a waiting loop until the g_signal "channel-event" occurs.
131  * If the event is SPICE_CHANNEL_OPENED, TRUE should be returned,
132  * otherwise FALSE should be returned.
133  */
134  return TRUE;
135 }

137 static gboolean remmina_plugin_spice_close_connection(RemminaProtocolWidget *gp)
138 {
139  TRACE_CALL(__func__);
140  RemminaPluginSpiceData *gpdata = GET_PLUGIN_DATA(gp);
141 
142  if (gpdata->main_channel) {
143  g_signal_handlers_disconnect_by_func(gpdata->main_channel,
144  G_CALLBACK(remmina_plugin_spice_main_channel_event_cb),
145  gp);
146  }
147 
148  if (gpdata->session) {
149  spice_session_disconnect(gpdata->session);
150  g_object_unref(gpdata->session);
151  gpdata->session = NULL;
152  remmina_plugin_service->protocol_plugin_emit_signal(gp, "disconnect");
153  }
154 
155 #ifdef SPICE_GTK_CHECK_VERSION
156 # if SPICE_GTK_CHECK_VERSION(0, 31, 0)
157  if (gpdata->file_transfers) {
158  g_hash_table_unref(gpdata->file_transfers);
159  }
160 # endif /* SPICE_GTK_CHECK_VERSION(0, 31, 0) */
161 #endif /* SPICE_GTK_CHECK_VERSION */
162 
163  return FALSE;
164 }

172  RemminaFile *remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
173 
174  g_object_get(channel, "channel-id", &id, NULL);
175 
176  if (SPICE_IS_MAIN_CHANNEL(channel)) {
177  gpdata->main_channel = SPICE_MAIN_CHANNEL(channel);
178  g_signal_connect(channel,
179  "channel-event",
180  G_CALLBACK(remmina_plugin_spice_main_channel_event_cb),
181  gp);
182 #ifdef SPICE_GTK_CHECK_VERSION
183 # if SPICE_GTK_CHECK_VERSION(0, 31, 0)
184  g_signal_connect(channel,

186  G_CALLBACK(remmina_plugin_spice_file_transfer_new_cb),
187  gp);
188 # endif /* SPICE_GTK_CHECK_VERSION(0, 31, 0) */
189 #endif /* SPICE_GTK_CHECK_VERSION */
190  }
191 
192  if (SPICE_IS_DISPLAY_CHANNEL(channel)) {
193  gpdata->display_channel = SPICE_DISPLAY_CHANNEL(channel);
194  gpdata->display = spice_display_new(gpdata->session, id);
195  g_signal_connect(gpdata->display,
196  "notify::ready",
197  G_CALLBACK(remmina_plugin_spice_display_ready_cb),
198  gp);
199  remmina_plugin_spice_display_ready_cb(G_OBJECT(gpdata->display), NULL, gp);
200  }
201 
202  if (SPICE_IS_PLAYBACK_CHANNEL(channel)) {
203  if (remmina_plugin_service->file_get_int(remminafile, "enableaudio", FALSE)) {
204  gpdata->audio = spice_audio_get(gpdata->session, NULL);
205  }
206  }
207 
208  if (SPICE_IS_WEBDAV_CHANNEL(channel)) {
209  if (remmina_plugin_service->file_get_string(remminafile, "sharefolder")) {
210  spice_channel_connect(channel);
211  }
212  }

240 static void remmina_plugin_spice_main_channel_event_cb(SpiceChannel *channel, SpiceChannelEvent event, RemminaProtocolWidget *gp)
241 {
242  TRACE_CALL(__func__);
243 
244  gchar *server;
245  gint port;
246  RemminaFile *remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
247 
248  switch (event) {
249  case SPICE_CHANNEL_CLOSED:
250  remmina_plugin_service->get_server_port(remmina_plugin_service->file_get_string(remminafile, "server"),
251  XSPICE_DEFAULT_PORT,
252  &server,
253  &port);
254  remmina_plugin_service->protocol_plugin_set_error(gp, _("Disconnected from SPICE server %s."), server);
255  g_free(server);
256  remmina_plugin_spice_close_connection(gp);
257  break;
258  case SPICE_CHANNEL_OPENED:
259  break;
260  case SPICE_CHANNEL_ERROR_AUTH:
261  if (remmina_plugin_spice_ask_auth(gp)) {
262  remmina_plugin_spice_open_connection(gp);
263  }else{
264  remmina_plugin_service->protocol_plugin_set_error(gp, _("Invalid password."));
265  remmina_plugin_spice_close_connection(gp);
266  }
267  break;
268  case SPICE_CHANNEL_ERROR_TLS:
269  remmina_plugin_service->protocol_plugin_set_error(gp, _("TLS connection error."));
270  remmina_plugin_spice_close_connection(gp);
271  break;
272  case SPICE_CHANNEL_ERROR_IO:
273  case SPICE_CHANNEL_ERROR_LINK:
274  case SPICE_CHANNEL_ERROR_CONNECT:
275  remmina_plugin_service->protocol_plugin_set_error(gp, _("Connection to SPICE server failed."));
276  remmina_plugin_spice_close_connection(gp);
277  break;
278  default:
279  break;
280  }
281 }

333 static void remmina_plugin_spice_update_scale(RemminaProtocolWidget *gp)
334 {
335  TRACE_CALL(__func__);
336 
337  gint scale, width, height;
338  RemminaPluginSpiceData *gpdata = GET_PLUGIN_DATA(gp);
339  RemminaFile *remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
340 
341  scale = remmina_plugin_service->file_get_int(remminafile, "scale", FALSE);
342  g_object_set(gpdata->display, "scaling", scale, NULL);
343 
344  if (scale) {
345  /* In scaled mode, the SpiceDisplay will get its dimensions from its parent */
346  gtk_widget_set_size_request(GTK_WIDGET(gpdata->display), -1, -1 );
347  }else {
348  /* In non scaled mode, the plugins forces dimensions of the SpiceDisplay */
349  g_object_get(gpdata->display_channel,
350  "width", &width,
351  "height", &height,
352  NULL);
353  gtk_widget_set_size_request(GTK_WIDGET(gpdata->display), width, height);
354  }

364 static void remmina_plugin_spice_call_feature(RemminaProtocolWidget *gp, const RemminaProtocolFeature *feature)
365 {
366  TRACE_CALL(__func__);
367 
368  RemminaPluginSpiceData *gpdata = GET_PLUGIN_DATA(gp);
369  RemminaFile *remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
370 
371  switch (feature->id) {
372  case REMMINA_PLUGIN_SPICE_FEATURE_PREF_VIEWONLY:
373  g_object_set(gpdata->session,

382  NULL);
383  break;
384  case REMMINA_PLUGIN_SPICE_FEATURE_PREF_DISABLECLIPBOARD:
385  g_object_set(gpdata->gtk_session,
386  "auto-clipboard",
387  !remmina_plugin_service->file_get_int(remminafile, "disableclipboard", FALSE),
388  NULL);
389  break;
390  case REMMINA_PLUGIN_SPICE_FEATURE_SCALE:
391  remmina_plugin_spice_update_scale(gp);
392  break;
393  case REMMINA_PLUGIN_SPICE_FEATURE_TOOL_SENDCTRLALTDEL:
394  remmina_plugin_spice_send_ctrlaltdel(gp);
395  break;
396  case REMMINA_PLUGIN_SPICE_FEATURE_TOOL_USBREDIR:
