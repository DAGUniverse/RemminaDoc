Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

398  }

400  ret = gtk_tree_model_iter_next(tree, iter);

402  return match;

803 static void remmina_main_import_file_list(GSList *files)
804 {
805  TRACE_CALL(__func__);
806  GtkWidget *dlg;
807  GSList *element;
808  gchar *path;
809  RemminaFilePlugin *plugin;
810  GString *err;
811  RemminaFile *remminafile = NULL;
812  gboolean imported;
813 
814  err = g_string_new(NULL);
815  imported = FALSE;
816  for (element = files; element; element = element->next) {
817  path = (gchar*)element->data;
818  plugin = remmina_plugin_manager_get_import_file_handler(path);
819  if (plugin && (remminafile = plugin->import_func(path)) != NULL && remmina_file_get_string(remminafile, "name")) {
820  remmina_file_generate_filename(remminafile);
821  remmina_file_save(remminafile);
822  imported = TRUE;
823  }else {
824  g_string_append(err, path);
825  g_string_append_c(err, '\n');
826  }
827  if (remminafile) {
828  remmina_file_free(remminafile);
829  remminafile = NULL;
830  }
831  g_free(path);
832  }
833  g_slist_free(files);
834  if (err->len > 0) {
835  dlg = gtk_message_dialog_new(remminamain->window, GTK_DIALOG_MODAL, GTK_MESSAGE_ERROR, GTK_BUTTONS_OK,
836  _("Unable to import:\n%s"), err->str);
837  g_signal_connect(G_OBJECT(dlg), "response", G_CALLBACK(gtk_widget_destroy), NULL);
838  gtk_widget_show(dlg);
839  }
840  g_string_free(err, TRUE);
841  if (imported) {
842  remmina_main_load_files();
843  }
844 }
