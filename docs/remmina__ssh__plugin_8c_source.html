Remmina - The GTK+ Remote Desktop Client  1.2.0
Remmina is a remote desktop client written in GTK+, aiming to be useful for system administrators and travellers, who need to work with lots of remote computers in front of either large monitors or tiny netbooks. Remmina supports multiple network protocols in an integrated and consistent user interface. Currently RDP, VNC, NX, XDMCP and SSH are supported.

216 remmina_plugin_ssh_main_thread(gpointer data)
217 {
218  TRACE_CALL(__func__);
219  RemminaProtocolWidget *gp = (RemminaProtocolWidget*)data;
220  RemminaPluginSshData *gpdata;
221  RemminaFile *remminafile;
222  RemminaSSH *ssh;
223  RemminaSSHShell *shell = NULL;
224  gboolean cont = FALSE;
225  gint ret;
226  gchar *charset;
227  const gchar *saveserver;
228  const gchar *saveusername;
229  gchar *hostport;
230  gchar tunneluser[33];
231  gchar tunnelserver[256];
232  gchar tunnelport[6];
233  gchar *host;
234  gint port;
235 
236  pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, NULL);
237  CANCEL_ASYNC
238 
239  gpdata = GET_PLUGIN_DATA(gp);
240 
248  remminafile = remmina_plugin_service->protocol_plugin_get_file(gp);
249  /* We save the ssh server name, so that we can restore it at the end of the connection */
250  saveserver = remmina_plugin_service->file_get_string(remminafile, "ssh_server");
251  remmina_plugin_service->file_set_string(remminafile, "save_ssh_server", g_strdup(saveserver));
252  /* We save the ssh username, so that we can restore it at the end of the connection */
253  saveusername = remmina_plugin_service->file_get_string(remminafile, "ssh_username");
254  remmina_plugin_service->file_set_string(remminafile, "save_ssh_username", g_strdup(saveusername));
255 
256  if (saveserver) {
260  if (strchr(saveserver, '@')) {
261  sscanf(saveserver, "%[_a-zA-Z0-9.]@%[_a-zA-Z0-9.]:%[0-9]",
262  tunneluser, tunnelserver, tunnelport);
263  g_print ("Username: %s, tunneluser: %s\n",
264  remmina_plugin_service->file_get_string(remminafile, "ssh_username"), tunneluser);
265  if (saveusername != NULL && tunneluser[0] != '\0') {
266  remmina_plugin_service->file_set_string(remminafile, "ssh_username", NULL);
267  }
268  remmina_plugin_service->file_set_string(remminafile, "ssh_username",
269  g_strdup(tunneluser));
270  remmina_plugin_service->file_set_string(remminafile, "ssh_server",
271  g_strconcat (tunnelserver, ":", tunnelport, NULL));
272  }
273  }
274 
275  hostport = remmina_plugin_service->protocol_plugin_start_direct_tunnel(gp, 22, FALSE);
276  /* We restore the ssh username as the tunnel is set */
277  remmina_plugin_service->file_set_string(remminafile, "ssh_username", g_strdup(saveusername));
278  if (hostport == NULL) {
279  return FALSE;
280  }
281  remmina_plugin_service->get_server_port(hostport, 22, &host, &port);
282 
283  ssh = g_object_get_data(G_OBJECT(gp), "user-data");
284  if (ssh) {
285  /* Create SSH Shell connection based on existing SSH session */
286  shell = remmina_ssh_shell_new_from_ssh(ssh);
287  if (remmina_ssh_init_session(REMMINA_SSH(shell)) &&
288  remmina_ssh_auth(REMMINA_SSH(shell), NULL) > 0 &&
289  remmina_ssh_shell_open(shell, (RemminaSSHExitFunc)
290  remmina_plugin_service->protocol_plugin_close_connection, gp)) {
291  cont = TRUE;
292  }
293  }else {
294  /* New SSH Shell connection */
295  if (remmina_plugin_service->file_get_int(remminafile, "ssh_enabled", FALSE)) {
296  remmina_plugin_service->file_set_string(remminafile, "ssh_server", g_strdup(hostport));
297  }else {
298  remmina_plugin_service->file_set_string(remminafile, "ssh_server",
299  remmina_plugin_service->file_get_string(remminafile, "server"));
300  }
301  g_free(hostport);
302  g_free(host);
303 
304  shell = remmina_ssh_shell_new_from_file(remminafile);
305  while (1) {
306  if (!remmina_ssh_init_session(REMMINA_SSH(shell))) {
307  remmina_plugin_service->protocol_plugin_set_error(gp, "%s", REMMINA_SSH(shell)->error);
308  break;
309  }
310 
311  ret = remmina_ssh_auth_gui(REMMINA_SSH(shell),
312  REMMINA_INIT_DIALOG(remmina_protocol_widget_get_init_dialog(gp)),
313  remminafile);
314  if (ret == 0) {
315  remmina_plugin_service->protocol_plugin_set_error(gp, "%s", REMMINA_SSH(shell)->error);
316  }
317  if (ret <= 0) break;
318 
319  if (!remmina_ssh_shell_open(shell, (RemminaSSHExitFunc)
320  remmina_plugin_service->protocol_plugin_close_connection, gp)) {
321  remmina_plugin_service->protocol_plugin_set_error(gp, "%s", REMMINA_SSH(shell)->error);
322  break;
323  }
324 
325  cont = TRUE;
326  break;
327  }
328 
329  /* We restore the ssh_server name */
330  remmina_plugin_service->file_set_string(remminafile, "ssh_server",
331  remmina_plugin_service->file_get_string(remminafile, "save_ssh_server"));
332  }
333  if (!cont) {
334  if (shell) remmina_ssh_shell_free(shell);
335  IDLE_ADD((GSourceFunc)remmina_plugin_service->protocol_plugin_close_connection, gp);
336  return NULL;
337  }
338 
339  gpdata->shell = shell;
340 
341  charset = REMMINA_SSH(shell)->charset;
342  remmina_plugin_ssh_vte_terminal_set_encoding_and_pty(VTE_TERMINAL(gpdata->vte), charset, shell->master, shell->slave);
343  remmina_plugin_service->protocol_plugin_emit_signal(gp, "connect");
344 
345  gpdata->thread = 0;
346  return NULL;
347 }
